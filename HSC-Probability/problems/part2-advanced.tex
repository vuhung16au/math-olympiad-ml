\begin{problem}[No sixes in repeated rolls]
A fair die is rolled $6$ times. Find the probability that no six appears.
\end{problem}

\begin{hint}
This is repeated independent failure with probability $\frac56$ each time.
\end{hint}

\begin{solution}
\[
P(\text{no sixes})=\left(\frac56\right)^6=\frac{15625}{46656}.
\]
\end{solution}

\begin{takeaways}
\item Repeated independent failures multiply.
\item ``No sixes'' means using the failure probability $\frac56$ on every roll.
\end{takeaways}

\begin{problem}[At most two successes]
If
\[
X\sim \mathrm{Bin}(4,0.3),
\]
find $P(X\le 2)$.
\end{problem}

\begin{hint}
Either add $P(X=0)$, $P(X=1)$, and $P(X=2)$, or subtract the complement.
\end{hint}

\begin{solution}
\[
P(X\le 2)=1-[P(X=3)+P(X=4)].
\]
Now
\[
P(X=3)=\binom43(0.3)^3(0.7)=4(0.027)(0.7)=0.0756,
\]
and
\[
P(X=4)=(0.3)^4=0.0081.
\]
Hence
\[
P(X\le 2)=1-0.0756-0.0081=0.9163.
\]
\end{solution}

\begin{takeaways}
\item For tail probabilities, complements often reduce the amount of arithmetic.
\item In binomial questions, choose the shorter side of the distribution to sum.
\end{takeaways}

\begin{problem}[Game value]
A game costs \$3 to play. A fair die is rolled. If a $6$ appears, the player receives \$10. Otherwise the player receives nothing. Find the expected profit for the player.
\end{problem}

\begin{hint}
Define profit, not payout.
\end{hint}

\begin{solution}
Let $X$ be the player's profit.

If a $6$ appears, profit is
\[
10-3=7
\]
with probability $\frac16$.

Otherwise profit is
\[
-3
\]
with probability $\frac56$.

So
\[
E(X)=7\left(\frac16\right)+(-3)\left(\frac56\right)
=\frac76-\frac{15}{6}
=-\frac86=-\frac43.
\]

Thus the expected profit is
\[
\boxed{-\frac43\text{ dollars}}.
\]
\end{solution}

\begin{takeaways}
\item Expected profit is not the same as expected payout.
\item Always define the random variable carefully before finding its mean.
\end{takeaways}

\begin{problem}[Variance from a small distribution]
A discrete random variable $X$ has
\[
P(X=1)=0.2,\qquad P(X=3)=0.5,\qquad P(X=5)=0.3.
\]
Find $\mathrm{Var}(X)$.
\end{problem}

\begin{hint}
Compute $E(X)$ and $E(X^2)$ separately.
\end{hint}

\begin{solution}
\[
E(X)=1(0.2)+3(0.5)+5(0.3)=0.2+1.5+1.5=3.2.
\]

\[
E(X^2)=1^2(0.2)+3^2(0.5)+5^2(0.3)=0.2+4.5+7.5=12.2.
\]

Therefore,
\[
\mathrm{Var}(X)=12.2-(3.2)^2=12.2-10.24=1.96.
\]
\end{solution}

\begin{takeaways}
\item Variance measures spread, so it is built from both the first and second moments.
\item A clean table method is $E(X^2)-[E(X)]^2$.
\end{takeaways}

\begin{problem}[Exactly three correct guesses]
A student guesses on $7$ true-false questions independently. Find the probability of getting exactly $3$ correct.
\end{problem}

\begin{hint}
This is binomial with $p=\frac12$.
\end{hint}

\begin{solution}
\[
P(X=3)=\binom73\left(\frac12\right)^7=35\cdot \frac1{128}=\frac{35}{128}.
\]
\end{solution}

\begin{takeaways}
\item A true-false guessing model is binomial with $p=\frac12$.
\item The combination counts where the $3$ correct answers occur.
\end{takeaways}

\begin{problem}[Expected number of red balls]
A box contains $5$ red balls and $7$ white balls. Three balls are drawn with replacement. Let $X$ be the number of red balls drawn. Find $E(X)$ and $\mathrm{Var}(X)$.
\end{problem}

\begin{hint}
With replacement, the number of reds is binomial.
\end{hint}

\begin{solution}
Here
\[
X\sim \mathrm{Bin}\left(3,\frac5{12}\right).
\]
Therefore
\[
E(X)=np=3\cdot \frac5{12}=\frac54,
\]
and
\[
\mathrm{Var}(X)=np(1-p)=3\cdot \frac5{12}\cdot \frac7{12}=\frac{35}{48}.
\]
\end{solution}

\begin{takeaways}
\item With replacement, the trials are independent, so a binomial model is appropriate.
\item Mean and variance both come from the same success probability $p=\frac5{12}$.
\end{takeaways}

\begin{problem}[A fair game check]
A game pays \$12 with probability $0.15$, \$5 with probability $0.25$, and \$0 otherwise. What entry fee would make the game fair?
\end{problem}

\begin{hint}
A fair game has expected profit $0$, so the fee must equal expected payout.
\end{hint}

\begin{solution}
The expected payout is
\[
12(0.15)+5(0.25)+0(0.60)=1.8+1.25=3.05.
\]
So the fair entry fee is
\[
\boxed{\$3.05}.
\]
\end{solution}

\begin{takeaways}
\item A fair game has expected profit $0$, so the fee must equal the expected payout.
\item Fair in the mathematical sense does not mean players always break even in practice.
\end{takeaways}

\begin{problem}[More than half successes]
If
\[
X\sim \mathrm{Bin}(5,0.2),
\]
find $P(X>2)$.
\end{problem}

\begin{hint}
Add $P(X=3)$, $P(X=4)$, and $P(X=5)$.
\end{hint}

\begin{solution}
\[
P(X>2)=P(X=3)+P(X=4)+P(X=5).
\]
Now
\[
P(X=3)=\binom53(0.2)^3(0.8)^2=10(0.008)(0.64)=0.0512,
\]
\[
P(X=4)=\binom54(0.2)^4(0.8)=5(0.0016)(0.8)=0.0064,
\]
\[
P(X=5)=(0.2)^5=0.00032.
\]
Therefore
\[
P(X>2)=0.0512+0.0064+0.00032=0.05792.
\]
\end{solution}

\begin{takeaways}
\item ``More than half'' becomes a right-tail binomial probability.
\item Writing the relevant values first helps avoid missing a term.
\end{takeaways}

\begin{problem}[Conditional probability after a draw]
A box contains $4$ defective bulbs and $9$ good bulbs. Two bulbs are drawn without replacement. Given that the first bulb drawn is defective, find the probability that the second bulb is good.
\end{problem}

\begin{hint}
Condition on the first draw and update the box contents.
\end{hint}

\begin{solution}
If the first bulb is defective, then $3$ defective and $9$ good bulbs remain, so $12$ bulbs remain in total.
Hence
\[
P(\text{second good}\mid \text{first defective})=\frac9{12}=\frac34.
\]
\end{solution}

\begin{takeaways}
\item Once a condition is given, update the contents before calculating.
\item Conditional probability after a draw is often just a revised fraction.
\end{takeaways}

\begin{problem}[Geometric probability in a rectangular field]
An $11\text{ m}\times 16\text{ m}$ rectangular field is considered, and a sheep wanders randomly around the field.

Find the probability that the sheep is:
\begin{enumerate}[(a)]
  \item more than $2\text{ m}$ from the edge of the field,
\end{enumerate}

\noindent\textit{Note.} This is a geometric probability question. If every point in the field is equally likely, then
\[
P(\text{event})=\frac{\text{favourable area}}{\text{total area}}.
\]

\begin{center}
\begin{tikzpicture}[x=0.38cm,y=0.38cm]
  \fill[softivory] (0,0) rectangle (16,11);
  \draw[line width=0.9pt] (0,0) rectangle (16,11);

  % Inner rectangle: more than 2 m from every edge.
  \fill[bookpurple!16] (2,2) rectangle (14,9);
  \draw[bookpurple, line width=0.8pt] (2,2) rectangle (14,9);

  % Dimension labels
  \node[below] at (8,0) {$16\text{ m}$};
  \node[rotate=90] at (16.9,5.5) {$11\text{ m}$};

  % Distance markers
  \draw[<->, bookpurple, line width=0.7pt] (0,1.2) -- (2,1.2);
  \node[bookpurple, below] at (1,1.2) {\scriptsize $2\text{ m}$};
  \draw[<->, bookpurple, line width=0.7pt] (1.2,0) -- (1.2,2);

  \node[bookpurple] at (8,5.5) {\scriptsize more than $2\text{ m}$ from every edge};
\end{tikzpicture}
\end{center}
\end{problem}

\begin{hint}
For part (a), imagine shrinking the rectangle by $2\text{ m}$ from every edge.
\end{hint}

\begin{solution}
The total area of the field is
\[
11\times 16=176\text{ m}^2.
\]

\textbf{(a)} Being more than $2\text{ m}$ from the edge means being inside the inner rectangle left after removing a $2\text{ m}$ strip from each side.

Its dimensions would be
\[
11-4=7
\qquad \text{and} \qquad
16-4=12,
\]
so its area is
\[
7\times 12=84\text{ m}^2.
\]

Therefore,
\[
P(\text{more than }2\text{ m from the edge})=\frac{84}{176}=\frac{21}{44}.
\]
\end{solution}

\begin{takeaways}
\item Geometric probability replaces counting by comparing favourable area with total area.
\item When a point must stay a fixed distance from every edge, the favourable region is an inner rectangle.
\end{takeaways}

\begin{problem}[Coin on a square grid]
A coin with radius $1\text{ cm}$ is thrown down on a square grid whose lines are $6\text{ cm}$ apart. Find the probability that the coin does not intersect any grid line.
\end{problem}

\noindent\textit{Note.} This is a geometric probability question. By symmetry, it is enough to study the position of the centre of the coin inside one $6\text{ cm}\times 6\text{ cm}$ square.

\begin{hint}
The coin avoids all grid lines exactly when its centre lands more than $1\text{ cm}$ away from every side of the square.
\end{hint}

\begin{solution}
Consider one square cell of the grid, with side length $6$ cm. The centre of the coin is equally likely to land anywhere in this square.

For the coin not to touch a grid line, its centre must be at least $1$ cm from each side. So the centre must lie inside the inner square obtained by removing a strip of width $1$ cm from each edge.

That inner square has side length
\[
6-2(1)=4,
\]
so its area is
\[
4^2=16.
\]

The total area of the square cell is
\[
6^2=36.
\]

Therefore
\[
P(\text{coin does not intersect a grid line})
=\frac{16}{36}
=\frac49.
\]
\end{solution}

\begin{takeaways}
\item In grid problems, track the position of the centre rather than the whole object.
\item The probability comes from the area of the safe inner region inside one cell.
\end{takeaways}

\begin{problem}[HSC 2024-style: Sample size from standard deviation]
Suppose $\hat p$ is the sample proportion of successes from $n$ independent Bernoulli trials with success probability $p=0.5$.

Find the smallest value of $n$ such that the standard deviation of $\hat p$ is less than $0.06$.
\end{problem}

\begin{hint}
Use
\[
\mathrm{Var}(\hat p)=\frac{p(1-p)}{n}.
\]
\end{hint}

\begin{solution}
Since $p=0.5$,
\[
\mathrm{Var}(\hat p)=\frac{0.5(0.5)}{n}=\frac{0.25}{n}.
\]
So
\[
\sigma_{\hat p}=\sqrt{\frac{0.25}{n}}=\frac{0.5}{\sqrt n}.
\]

We require
\[
\frac{0.5}{\sqrt n}<0.06.
\]
Hence
\[
\sqrt n>\frac{0.5}{0.06}=\frac{25}{3},
\]
so
\[
n>\frac{625}{9}\approx 69.44.
\]

Therefore the smallest integer value is
\[
\boxed{70}.
\]
\end{solution}

\begin{takeaways}
\item Sample proportion questions often reduce to controlling the standard deviation.
\item After solving the inequality, round up because the sample size must be an integer.
\end{takeaways}

\begin{problem}[HSC 2024-style: Charity donations]
A charity worker talks to exactly $100$ people in a day. The probability that any given person makes a donation is $0.31$, independently of the others.

Use a normal approximation to estimate the probability that at least $35$ people make a donation.
\end{problem}

\begin{hint}
Model the number of donations as binomial, then approximate with a normal distribution and use a continuity correction.
\end{hint}

\begin{solution}
Let $X$ be the number of donations. Then
\[
X\sim \mathrm{Bin}(100,0.31).
\]

So
\[
\mu=np=31,
\qquad
\sigma^2=np(1-p)=100(0.31)(0.69)=21.39.
\]
Thus
\[
\sigma\approx 4.63.
\]

We want
\[
P(X\ge 35).
\]
Using a continuity correction,
\[
P(X\ge 35)\approx P(Y\ge 34.5),
\]
where
\[
Y\sim N(31,21.39).
\]

Then
\[
P(X\ge 35)\approx P\left(Z\ge \frac{34.5-31}{4.63}\right)
=P(Z\ge 0.76).
\]

Hence
\[
P(X\ge 35)\approx 1-0.7764=0.2236.
\]

So the required probability is approximately
\[
\boxed{0.22}.
\]
\end{solution}

\begin{takeaways}
\item A normal approximation needs the binomial mean and variance first.
\item Continuity correction is part of the model, not an optional decoration.
\end{takeaways}

\begin{problem}[Designing the ``Lucky 3'' lottery game]
A local club is designing a new lottery game called \textit{Lucky 3}. A player selects $3$ distinct numbers from
\[
\{1,2,3,4,5,6,7,8,9\},
\]
and the host then draws $3$ distinct winning numbers from the same set.

Tier 1 pays when all $3$ numbers match, in any order. Tier 2 pays when exactly $2$ numbers match.

\begin{enumerate}[(i)]
  \item Show that the total number of possible selections is $84$.
  \item Let $X$ be the number of matches. Find $P(X=3)$ and show that
  \[
  P(X=2)=\frac{3}{14}.
  \]
  \item If the prizes are \$40 for $3$ matches, \$10 for exactly $2$ matches, and \$0 otherwise, find the expected payout for a single game.
  \item If the ticket price is \$5, find the expected profit per ticket for the club and the expected total profit on $1000$ tickets.
  \item Explain why increasing the Tier 2 prize from \$10 to \$15 could be risky for the club, even if the game still has positive expected profit.
\end{enumerate}
\end{problem}

\begin{hint}
Use combinations for the matching probabilities, then separate \emph{expected payout} from \emph{expected profit}. For the final part, think about how a smaller margin affects risk.
\end{hint}

\begin{solution}
\textbf{(i)} The total number of selections is
\[
\binom93=84.
\]

\textbf{(ii)} For $3$ matches,
\[
P(X=3)=\frac1{84}.
\]
For exactly $2$ matches,
\[
P(X=2)=\frac{\binom32\binom61}{\binom93}
=\frac{18}{84}
=\frac{3}{14}.
\]

\textbf{(iii)} If $P$ is the payout, then
\[
E(P)=40\left(\frac1{84}\right)+10\left(\frac3{14}\right)
=\frac{220}{84}
=\frac{55}{21}
\approx \$2.62.
\]

\textbf{(iv)} Expected profit per ticket:
\[
5-\frac{55}{21}=\frac{50}{21}\approx \$2.38.
\]
For $1000$ tickets, the expected total profit is
\[
1000\cdot \frac{50}{21}\approx \$2381.
\]

\textbf{(v)} If the Tier 2 prize becomes \$15, then
\[
E(P)=40\left(\frac1{84}\right)+15\left(\frac3{14}\right)
=\frac{310}{84}
\approx \$3.69.
\]
So the expected profit per ticket drops to about
\[
5-3.69=\$1.31.
\]
The game still favours the club on average, but the profit margin is much smaller, so a run of extra Tier 2 winners would create more financial pressure. The game becomes riskier even though the mean profit remains positive.
\end{solution}

\begin{takeaways}
\item Expected payout and expected profit are different quantities and should be kept separate.
\item A positive mean profit does not remove short-run financial risk when prizes are volatile.
\end{takeaways}

\begin{problem}[The truncated St Petersburg game]
A fair coin is tossed until a head appears. If the first head appears on toss number $k$, the prize is
\[
W=2^k \text{ dollars.}
\]
The casino caps the game at $n$ tosses. If no head has appeared by the $n$th toss, the game ends and the player is paid the maximum prize
\[
2^n.
\]

\begin{enumerate}[(i)]
  \item Suppose $n=10$.
  \begin{enumerate}[(a)]
    \item List the possible values of $W$ and their probabilities.
    \item Show that the expected payout is \$11.
    \item If the entry fee is \$40, find the casino's expected profit per game.
  \end{enumerate}
  \item Let $E_n$ be the expected payout for a game capped at $n$ tosses.
  Show that $E_n=n+1$, then find the value of $n$ for which the expected payout is exactly \$40 and the corresponding maximum jackpot in scientific notation.
  \item Using the version with expected payout \$40:
  \begin{enumerate}[(a)]
    \item Find the probability that a player wins more than the \$40 entry fee.
    \item Comment on the difference between expected value and the most likely outcome for the player.
  \end{enumerate}
\end{enumerate}
\end{problem}

\begin{hint}
For the capped game, the first $n-1$ payouts have the usual probabilities $\left(\frac12\right)^k$, but the final payout $2^n$ includes both ``first head on toss $n$'' and ``no head in the first $n$ tosses''.
\end{hint}

\begin{solution}
\textbf{(i)} For $n=10$, the payouts are $2,4,8,\dots,2^{10}$. For $k=1,\dots,9$,
\[
P(W=2^k)=\left(\frac12\right)^k,
\qquad
P(W=2^{10})=\left(\frac12\right)^9.
\]
Hence
\[
E(W)=\sum_{k=1}^{9}2^k\left(\frac12\right)^k+2^{10}\left(\frac12\right)^9
=9+2=11,
\]
so the casino's expected profit at a \$40 entry fee is
\[
40-11=\$29.
\]

\textbf{(ii)} In general,
\[
E_n=\sum_{k=1}^{n-1}2^k\left(\frac12\right)^k+2^n\left(\frac12\right)^{n-1}
=(n-1)+2=n+1.
\]
So $E_n=40$ gives
\[
n+1=40 \implies n=39.
\]
The maximum jackpot is
\[
2^{39}=549755813888\approx 5.50\times 10^{11}\text{ dollars.}
\]

\textbf{(iii)} To win more than \$40, the player needs $2^k>40$, so $k\ge 6$. Therefore
\[
P(W>40)=P(\text{no head in the first 5 tosses})=\left(\frac12\right)^5=\frac1{32}\approx 3.1\%.
\]
So even when the expected payout is \$40, the player wins more than the entry fee only about $3.1\%$ of the time; the expectation is supported by a tiny chance of a huge jackpot, while the most likely practical outcome is a loss.
\end{solution}

\begin{takeaways}
\item Expected value can be dominated by very rare, very large outcomes.
\item This is a classic reminder that ``fair on average'' and ``likely to feel fair'' are not the same idea.
\end{takeaways}

\begin{problem}[HSC 2025-style: Inverse-tail green-counter problem]
A bag contains counters, some of which are green. One hundred trials are run. In each trial, one counter is selected at random, its colour is noted, and the counter is returned to the bag.

Let $X$ be the number of times that a green counter is selected. It is known that
\[
E(X)=20
\]
and
\[
P(X\ge k)=0.0668.
\]
Find the value of $k$ using a normal approximation.
\end{problem}

\begin{hint}
From $E(X)=20$ with $100$ trials, find $p$, then approximate the binomial distribution by a normal distribution and work backwards from the tail probability.
\end{hint}

\begin{solution}
Since
\[
E(X)=np=100p=20,
\]
we have
\[
p=0.2.
\]
So
\[
X\sim \mathrm{Bin}(100,0.2).
\]

Hence
\[
\mu=np=20,
\qquad
\sigma^2=np(1-p)=100(0.2)(0.8)=16,
\]
so
\[
\sigma=4.
\]

Now
\[
P(X\ge k)=0.0668.
\]
Using the standard normal distribution, this corresponds to an upper-tail $z$-value of about $1.5$, since
\[
P(Z\ge 1.5)\approx 0.0668.
\]

Using a continuity correction,
\[
\frac{k-0.5-20}{4}\approx 1.5.
\]
So
\[
k-0.5\approx 26
\]
and therefore
\[
k\approx 26.5.
\]

Hence the required integer value is
\[
\boxed{27}.
\]
\end{solution}

\begin{takeaways}
\item In inverse-tail questions, work backward from the normal tail to a $z$-score.
\item Continuity correction is important when converting the binomial threshold to a normal one.
\end{takeaways}

\begin{problem}[HSC 2022-style: Airline overbooking threshold]
An airline has $350$ seats on a flight. Historical data suggests that each booked passenger independently fails to show up with probability $0.05$.

The airline wants to sell the largest number $N$ of tickets such that the probability of more passengers arriving than there are seats is less than $0.01$.

Explain how this can be modelled, and identify the inequality that $N$ must satisfy.
\end{problem}

\begin{hint}
Model the number of passengers who arrive, or equivalently the number who do not show up.
\end{hint}

\begin{solution}
Let $X$ be the number of passengers who arrive. Then each passenger arrives with probability
\[
0.95,
\]
so
\[
X\sim \mathrm{Bin}(N,0.95).
\]

If more passengers arrive than there are seats, then
\[
X>350.
\]

So the airline wants the largest value of $N$ such that
\[
P(X>350)<0.01.
\]

Equivalently, if $Y$ is the number of no-shows, then
\[
Y\sim \mathrm{Bin}(N,0.05)
\]
and the condition becomes
\[
P(Y<N-350)<0.01.
\]

This sets up the threshold problem, which can then be solved approximately using a normal model.
\end{solution}

\begin{takeaways}
\item Modelling is the main goal here: define the random variable and write the correct threshold inequality.
\item Overbooking questions connect expectation ideas to decision-making under risk.
\end{takeaways}
