\begin{problem}[A Similar Sum-of-Squares Quartic]
Consider the polynomial
\[
P(x)=x^4-7x^2+6x+25.
\]
\begin{enumerate}[label=(\roman*)]
\item Show that $P(x)$ can be expressed in the form $(x^2-a)^2+(x-b)^2$.
\item How many $x$-intercepts does the graph of $P(x)$ have?
\end{enumerate}
\end{problem}

\begin{hintbox}
Expand $(x^2-a)^2+(x-b)^2$ and compare coefficients.
\end{hintbox}

\begin{solution}
Expand:
\[
(x^2-a)^2+(x-b)^2=x^4+(-2a+1)x^2-2bx+(a^2+b^2).
\]
Comparing with $x^4-7x^2+6x+25$ gives
\[
-2a+1=-7,\qquad -2b=6.
\]
So
\[
a=4,\qquad b=-3.
\]
Also
\[
a^2+b^2=16+9=25,
\]
so
\[
P(x)=(x^2-4)^2+(x+3)^2.
\]
As a sum of two squares, $P(x)\ge 0$ for all $x$, and equality would require
\[
x^2-4=0\quad\text{and}\quad x+3=0
\]
at the same time, which is impossible. Thus $P(x)>0$ for all $x$.
Hence the graph has
\[
\boxed{0\text{ x-intercepts}.}
\]
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item Writing a polynomial as a sum of squares is a strong non-negativity argument.
\item If the sum of squares can never be zero, the graph never meets the $x$-axis.
\end{itemize}
\end{takeaways}

\begin{problem}[Intersections of $(x+1)^n$ and $(x+1)^m$]
At what points do the graphs of
\[
f(x)=(x+1)^n,\qquad g(x)=(x+1)^m
\]
intersect, where $m\ne n$?
\end{problem}

\begin{hintbox}
Let $t=x+1$ and solve $t^n=t^m$.
\end{hintbox}

\begin{solution}
Set $t=x+1$. Then intersections satisfy
\[
t^n=t^m.
\]
If $t=0$, then both sides are $0$, giving
\[
x=-1 \Rightarrow (-1,0).
\]
If $t\ne 0$, then
\[
t^{n-m}=1.
\]
Over the reals this gives $t=1$, and sometimes $t=-1$.

\begin{itemize}[leftmargin=*]
\item $t=1 \Rightarrow x=0$, so $(0,1)$ is always an intersection.
\item $t=-1 \Rightarrow x=-2$. This works exactly when $(-1)^n=(-1)^m$, i.e. when $m$ and $n$ have the same parity.
\end{itemize}

Hence the graphs always intersect at
\[
\boxed{(-1,0)\text{ and }(0,1),}
\]
and they also intersect at $x=-2$ when $m$ and $n$ have the same parity. The corresponding point is $(-2,1)$ if both are even, and $(-2,-1)$ if both are odd.
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item The substitution $t=x+1$ exposes the real structure immediately.
\item In power equations, always check $t=0$, $1$, and $-1$.
\end{itemize}
\end{takeaways}

\begin{problem}[A Factor-Theorem Identity in Three Variables]
Use the factor theorem to prove that
\[
x+y+z
\]
is a factor of
\[
x^3+y^3+z^3-3xyz.
\]
Hence find the other factor.
\end{problem}

\begin{hintbox}
Treat the expression as a polynomial in $x$ and substitute $x=-(y+z)$.
\end{hintbox}

\begin{solution}
Let
\[
F(x)=x^3+y^3+z^3-3xyz,
\]
where $y$ and $z$ are constants. Then
\begin{align*}
F(-(y+z))
&=-(y+z)^3+y^3+z^3+3yz(y+z)\\
&=0.
\end{align*}
So $x+y+z$ is a factor.

The full factorisation is
\[
x^3+y^3+z^3-3xyz=(x+y+z)(x^2+y^2+z^2-xy-yz-zx).
\]
Hence the other factor is
\[
\boxed{x^2+y^2+z^2-xy-yz-zx.}
\]
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item The factor theorem also works when the coefficients involve other variables.
\item This is a standard factorisation worth remembering.
\end{itemize}
\end{takeaways}

\begin{problem}[A Common Quadratic Factor]
If $a$ and $b$ are non-zero and $a+b=0$, prove that
\[
A(x)=x^3+ax^2-x+b
\]
and
\[
B(x)=x^3+bx^2-x+a
\]
have a common factor of degree $2$ but are not identical. What is the common factor?
\end{problem}

\begin{hintbox}
Replace $b$ by $-a$ and factor by grouping.
\end{hintbox}

\begin{solution}
Since $a+b=0$, we have $b=-a$.
Then
\[
A(x)=x^3+ax^2-x-a=x^2(x+a)-1(x+a)=(x+a)(x^2-1),
\]
and
\[
B(x)=x^3-ax^2-x+a=x^2(x-a)-1(x-a)=(x-a)(x^2-1).
\]
Therefore the common quadratic factor is
\[
\boxed{x^2-1.}
\]
They are not identical because $a\ne0$, so their linear factors are different.
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item A condition like $a+b=0$ should be used immediately.
\item Grouping is often enough to reveal a hidden common factor.
\end{itemize}
\end{takeaways}

\begin{problem}[Intersections via $P(x)-Q(x)$]
By factoring
\[
F(x)=P(x)-Q(x),
\]
describe the intersections between the curves
\[
P(x)=x^4+2x^3+4x-1
\]
and
\[
Q(x)=x^4+3x^2+5x-4.
\]
\end{problem}

\begin{hintbox}
The curves intersect exactly when $P(x)-Q(x)=0$.
\end{hintbox}

\begin{solution}
\[
F(x)=P(x)-Q(x)=2x^3-3x^2-x+3.
\]
Factor by grouping:
\[
2x^3-3x^2-x+3=x^2(2x-3)-1(2x-3)=(x^2-1)(2x-3).
\]
So the intersection $x$-coordinates are
\[
x=-1,\quad x=1,\quad x=\frac32.
\]
Now
\[
P(-1)=Q(-1)=-4,\qquad P(1)=Q(1)=6,\qquad P\!\left(\frac32\right)=Q\!\left(\frac32\right)=\frac{77}{16}.
\]
Hence the curves intersect at
\[
\boxed{(-1,-4),\ \left(1,6\right),\ \left(\frac32,\frac{77}{16}\right).}
\]
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item To compare two curves, subtract first and solve one equation.
\item The factorisation of the difference controls the number of intersections.
\end{itemize}
\end{takeaways}

\begin{problem}[The Reversal Transformation]
Consider the cubic equation
\[
x^3+ax^2+bx+c=0,
\]
where $c\ne0$, with roots $\alpha,\beta,\gamma$.
\begin{enumerate}[label=(\roman*)]
\item By using the substitution $y=\frac{1}{x}$, find a cubic equation in $y$ whose roots are $\frac1\alpha,\frac1\beta,\frac1\gamma$.
\item Hence, or otherwise, find the roots of
\[
cx^3-bx^2+ax-1=0
\]
in terms of $\alpha,\beta,\gamma$.
\end{enumerate}
\end{problem}

\begin{hintbox}
First substitute $x=\frac1y$. Then compare your new equation with the target equation by replacing the variable with its negative.
\end{hintbox}

\begin{solution}
\textbf{(i)} Let
\[
y=\frac1x,
\]
so
\[
x=\frac1y.
\]
Substitute into
\[
x^3+ax^2+bx+c=0:
\]
\[
\left(\frac1y\right)^3+a\left(\frac1y\right)^2+b\left(\frac1y\right)+c=0.
\]
Thus
\[
\frac1{y^3}+\frac{a}{y^2}+\frac{b}{y}+c=0.
\]
Multiply by $y^3$:
\[
1+ay+by^2+cy^3=0.
\]
So the required cubic is
\[
\boxed{cy^3+by^2+ay+1=0,}
\]
whose roots are
\[
\boxed{\frac1\alpha,\ \frac1\beta,\ \frac1\gamma.}
\]

\textbf{(ii)} From part (i), the equation
\[
cx^3+bx^2+ax+1=0
\]
has roots
\[
\frac1\alpha,\ \frac1\beta,\ \frac1\gamma.
\]
Now replace $x$ by $-z$:
\[
c(-z)^3+b(-z)^2+a(-z)+1=0.
\]
This gives
\[
-cz^3+bz^2-az+1=0.
\]
Multiply by $-1$:
\[
cz^3-bz^2+az-1=0.
\]
So the roots of the target equation are the negatives of the reciprocal roots. Therefore the roots are
\[
\boxed{-\frac1\alpha,\ -\frac1\beta,\ -\frac1\gamma.}
\]
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item The substitution $x=\frac1y$ reverses the order of the coefficients.
\item The substitution $x=-y$ flips the signs of the odd-power terms.
\item When a new polynomial looks like a reversed or sign-altered version of an old one, a substitution is usually the intended method.
\end{itemize}
\end{takeaways}

\begin{problem}[The Cosine-Polynomial Bridge]
Use the identity
\[
\cos(3\theta)=4\cos^3\theta-3\cos\theta,
\]
but do not prove it.

Start with
\[
\cos(3\theta)=\frac12.
\]
Rewrite this equation using only $\cos\theta$. Then let
\[
x=\cos\theta
\]
and show that the equation becomes
\[
8x^3-6x-1=0.
\]

Hence show that:
\begin{enumerate}[label=(\roman*)]
\item $\cos\frac{\pi}{9}=\cos\frac{2\pi}{9}+\cos\frac{4\pi}{9}$
\item $\sec\frac{\pi}{9}\sec\frac{2\pi}{9}\sec\frac{4\pi}{9}=8$
\item $\sec^2\frac{\pi}{9}+\sec^2\frac{2\pi}{9}+\sec^2\frac{4\pi}{9}=36$
\end{enumerate}
\end{problem}

\begin{hintbox}
Solve $\cos(3\theta)=\frac12$ first. Then rewrite obtuse-angle cosines using $\cos(\pi-\alpha)=-\cos\alpha$ before applying Vieta.
\end{hintbox}

\begin{solution}
From
\[
\cos(3\theta)=4\cos^3\theta-3\cos\theta=\frac12,
\]
we get
\[
8\cos^3\theta-6\cos\theta-1=0.
\]
Let $x=\cos\theta$. Then
\[
\boxed{8x^3-6x-1=0.}
\]

Now $\cos(3\theta)=\frac12$ gives
\[
3\theta=\frac{\pi}{3},\ \frac{5\pi}{3},\ \frac{7\pi}{3},
\]
so the roots are
\[
\cos\frac{\pi}{9},\ \cos\frac{5\pi}{9},\ \cos\frac{7\pi}{9}
=\cos\frac{\pi}{9},\ -\cos\frac{4\pi}{9},\ -\cos\frac{2\pi}{9}.
\]

\textbf{(i)} Since the $x^2$ coefficient is $0$,
\[
\cos\frac{\pi}{9}-\cos\frac{4\pi}{9}-\cos\frac{2\pi}{9}=0,
\]
hence
\[
\boxed{\cos\frac{\pi}{9}=\cos\frac{2\pi}{9}+\cos\frac{4\pi}{9}.}
\]

\textbf{(ii)} The product of the roots is
\[
-\frac{-1}{8}=\frac18,
\]
so
\[
\cos\frac{\pi}{9}\cos\frac{2\pi}{9}\cos\frac{4\pi}{9}=\frac18.
\]
Taking reciprocals,
\[
\boxed{\sec\frac{\pi}{9}\sec\frac{2\pi}{9}\sec\frac{4\pi}{9}=8.}
\]

\textbf{(iii)} For reciprocal roots, let $y=\frac1x$. Then
\[
8\left(\frac1y\right)^3-6\left(\frac1y\right)-1=0
\quad\Rightarrow\quad
y^3+6y^2-8=0.
\]
Its roots are
\[
\sec\frac{\pi}{9},\ -\sec\frac{4\pi}{9},\ -\sec\frac{2\pi}{9}.
\]
So
\[
\sum y=-6,\qquad \sum y_iy_j=0,
\]
and therefore
\[
\sum y^2=(\sum y)^2-2\sum y_iy_j=36.
\]
Hence
\[
\boxed{\sec^2\frac{\pi}{9}+\sec^2\frac{2\pi}{9}+\sec^2\frac{4\pi}{9}=36.}
\]
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item A multiple-angle identity can turn a trigonometric equation into a polynomial equation.
\item Once the trigonometric values are identified as roots, Vieta can produce elegant identities quickly.
\item Reciprocal-root substitutions are especially useful for secant and cosecant expressions.
\end{itemize}
\end{takeaways}

\begin{problem}[The Tangent Cubic]
It is known that
\[
\tan 3\theta=\frac{3\tan\theta-\tan^3\theta}{1-3\tan^2\theta}.
\]
\begin{enumerate}[label=(\roman*)]
\item By letting $x=\tan\theta$, solve the equation
\[
x^3-3x^2-3x+1=0.
\]
\item Hence find the exact value of
\[
\tan\frac{\pi}{12}+\tan\frac{5\pi}{12}.
\]
\end{enumerate}
\end{problem}

\begin{hintbox}
Rewrite the cubic so that the numerator and denominator in the $\tan 3\theta$ identity appear naturally. Then use Vieta for part (ii).
\end{hintbox}

\begin{solution}
\textbf{(i)} Let
\[
x=\tan\theta.
\]
Then
\[
\tan^3\theta-3\tan^2\theta-3\tan\theta+1=0.
\]
Rearrange:
\[
3\tan\theta-\tan^3\theta=1-3\tan^2\theta.
\]
So
\[
\frac{3\tan\theta-\tan^3\theta}{1-3\tan^2\theta}=1,
\]
hence
\[
\tan 3\theta=1.
\]
Take three distinct solutions:
\[
3\theta=\frac{\pi}{4},\ \frac{5\pi}{4},\ \frac{9\pi}{4}.
\]
Thus
\[
\theta=\frac{\pi}{12},\ \frac{5\pi}{12},\ \frac{3\pi}{4}.
\]
Therefore the roots are
\[
\boxed{\tan\frac{\pi}{12},\ \tan\frac{5\pi}{12},\ \tan\frac{3\pi}{4}}
\]
that is,
\[
\boxed{\tan\frac{\pi}{12},\ \tan\frac{5\pi}{12},\ -1.}
\]

\textbf{(ii)} By Vieta, the sum of the roots of
\[
x^3-3x^2-3x+1=0
\]
is
\[
-\frac{-3}{1}=3.
\]
So
\[
\tan\frac{\pi}{12}+\tan\frac{5\pi}{12}+(-1)=3.
\]
Hence
\[
\boxed{\tan\frac{\pi}{12}+\tan\frac{5\pi}{12}=4.}
\]
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item Some cubics are designed to match a trigonometric identity after a smart substitution.
\item Once the roots are known in trigonometric form, Vieta can evaluate sums very quickly.
\item A simple root like $-1$ is a useful check that the transformed solutions make sense.
\end{itemize}
\end{takeaways}

\begin{problem}[The Fibonacci Polynomial Remainder]
Consider the quadratic polynomial
\[
P(x)=x^2-x-1.
\]
Let $\alpha$ and $\beta$ be the roots of $P(x)=0$, where $\alpha$ is the positive root.

\begin{enumerate}[label=(\roman*)]
\item Using the Polynomial Division Algorithm, explain why for any integer $n\ge 1$, we can write
\[
x^n=Q_n(x)(x^2-x-1)+(F_nx+G_n),
\]
where $Q_n(x)$ is a polynomial and $F_n,G_n$ are integers.
\item By substituting $x=\alpha$, show that
\[
\alpha^n=F_n\alpha+G_n.
\]
\item Multiply both sides of the equation in part (ii) by $\alpha$. Using $\alpha^2=\alpha+1$, show that
\[
\alpha^{n+1}=(F_n+G_n)\alpha+F_n.
\]
\item Given that $F_n$ and $G_n$ are integers, and $\alpha=\frac{1+\sqrt5}{2}$ is irrational, deduce that
\[
F_{n+1}=F_n+G_n,\qquad G_{n+1}=F_n.
\]
Hence prove that
\[
F_{n+1}=F_n+F_{n-1}.
\]
\item Find the remainder when $x^{2026}$ is divided by $x^2-x-1$.
\end{enumerate}
\end{problem}

\begin{hintbox}
Treat $x^2=x+1$ as a reduction rule. For part (iv), compare two expressions of the form $A\alpha+B$ and use the irrationality of $\alpha$.
\end{hintbox}

\begin{solution}
\textbf{(i)} Dividing $x^n$ by the quadratic $x^2-x-1$, the remainder must have degree less than $2$. So
\[
x^n=Q_n(x)(x^2-x-1)+(F_nx+G_n)
\]
for some polynomial $Q_n(x)$ and constants $F_n,G_n$.

\textbf{(ii)} Since $\alpha$ is a root of $x^2-x-1=0$, substituting $x=\alpha$ gives
\[
\alpha^n=Q_n(\alpha)\cdot 0+F_n\alpha+G_n=F_n\alpha+G_n.
\]

\textbf{(iii)} Multiply by $\alpha$:
\[
\alpha^{n+1}=F_n\alpha^2+G_n\alpha.
\]
Using $\alpha^2=\alpha+1$,
\[
\alpha^{n+1}=F_n(\alpha+1)+G_n\alpha=(F_n+G_n)\alpha+F_n.
\]

\textbf{(iv)} But also
\[
\alpha^{n+1}=F_{n+1}\alpha+G_{n+1}.
\]
Comparing with part (iii),
\[
F_{n+1}\alpha+G_{n+1}=(F_n+G_n)\alpha+F_n.
\]
Since $\alpha$ is irrational and all coefficients are integers,
\[
F_{n+1}=F_n+G_n,\qquad G_{n+1}=F_n.
\]
Now replace $n$ by $n-1$ in the second relation:
\[
G_n=F_{n-1}.
\]
Hence
\[
\boxed{F_{n+1}=F_n+F_{n-1}.}
\]

\textbf{(v)} For $n=1$,
\[
x=1\cdot x+0,
\]
so $F_1=1$, $G_1=0$.
For $n=2$,
\[
x^2=x+1,
\]
so $F_2=1$, $G_2=1$.
From part (iv),
\[
F_{n+1}=F_n+F_{n-1}
\quad\text{and}\quad
G_n=F_{n-1}.
\]
Therefore the remainder when $x^{2026}$ is divided by $x^2-x-1$ is
\[
\boxed{F_{2026}x+F_{2025}.}
\]
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item Dividing by a quadratic always leaves a linear remainder.
\item A polynomial relation such as $x^2=x+1$ can reduce high powers recursively.
\item The golden ratio, Fibonacci numbers, and polynomial remainders are tightly connected by the equation $x^2-x-1=0$.
\item \textit{Math is Interconnected}: The golden ratio, polynomial remainders, and the Fibonacci sequence are all just different languages describing the exact same underlying mathematical structure: $x^2 - x - 1 = 0$.
\item \textit{Irrational Independence}: A core Extension 1/2 skill is recognizing that an equation of the form $a + b\sqrt{k} = c + d\sqrt{k}$ (where $a,b,c,d$ are rational and $\sqrt{k}$ is irrational) forces $a=c$ and $b=d$. We used this to equate the coefficients in part (iv).
\end{itemize}
\end{takeaways}

\begin{problem}[The Chebyshev Recurrence]

Chebyshev polynomials are arguably one of the most important families of polynomials in all of applied mathematics. While they start as a neat trigonometric trick in high school, they quickly become the backbone of modern computing and engineering.


This problem introduces the Chebyshev polynomials of the first kind, which are defined by a simple substitution from cosine functions.


Let $x=\cos\theta$ for $0\le \theta\le \pi$, so $x\in[-1,1]$. Define
\[
T_n(x)=\cos(n\theta)
\]
for integers $n\ge 0$.

\begin{enumerate}[label=(\roman*)]
\item Find $T_0(x)$ and $T_1(x)$. Using the double-angle formula, show that
\[
T_2(x)=2x^2-1.
\]
\item By expanding $\cos(n\theta+\theta)$ and $\cos(n\theta-\theta)$, prove that
\[
T_{n+1}(x)=2xT_n(x)-T_{n-1}(x).
\]
\item Use the recurrence to find $T_3(x)$ and $T_4(x)$.
\item \textbf{Engineering Application:} In telecommunications, a mathematical filter must stay bounded between $-1$ and $1$ inside the allowed frequency band, where $-1\le x\le 1$. But once the signal leaves this zone ($x>1$), the polynomial should grow as steeply as possible to help block interference.

Consider your Chebyshev polynomial
\[
T_3(x)=4x^3-3x
\]
and the standard cubic
\[
S(x)=x^3.
\]
Notice that both pass through $(1,1)$. By finding $T_3'(x)$ and $S'(x)$, compare their gradients at $x=1$. Hence show algebraically that for all $x>1$, the Chebyshev polynomial grows faster than the standard cubic.
\end{enumerate}
\end{problem}

\begin{hintbox}
Add the expansions of $\cos(n\theta+\theta)$ and $\cos(n\theta-\theta)$ so the sine terms cancel. For part (iv), compare
\[
T_3'(x)=12x^2-3
\qquad\text{and}\qquad
S'(x)=3x^2,
\]
then look at their difference for $x>1$.
\end{hintbox}

\begin{solution}
\textbf{(i)} Since
\[
T_0(x)=\cos 0=1,\qquad T_1(x)=\cos\theta=x,
\]
we have
\[
\boxed{T_0(x)=1,\qquad T_1(x)=x.}
\]
Also
\[
T_2(x)=\cos(2\theta)=2\cos^2\theta-1=2x^2-1.
\]

\textbf{(ii)} Using the sum and difference formulas,
\[
\cos(n\theta+\theta)=\cos(n\theta)\cos\theta-\sin(n\theta)\sin\theta,
\]
\[
\cos(n\theta-\theta)=\cos(n\theta)\cos\theta+\sin(n\theta)\sin\theta.
\]
Add them:
\[
\cos((n+1)\theta)+\cos((n-1)\theta)=2\cos(n\theta)\cos\theta.
\]
Substitute $T_n(x)=\cos(n\theta)$ and $x=\cos\theta$:
\[
T_{n+1}(x)+T_{n-1}(x)=2xT_n(x),
\]
so
\[
\boxed{T_{n+1}(x)=2xT_n(x)-T_{n-1}(x).}
\]

\textbf{(iii)} Apply the recurrence:
\[
T_3(x)=2x(2x^2-1)-x=4x^3-3x,
\]
\[
T_4(x)=2x(4x^3-3x)-(2x^2-1)=8x^4-8x^2+1.
\]

\textbf{(iv)} Left as a lovely challenge problem. It gives a nice glimpse of why Chebyshev polynomials matter so much in engineering and signal processing.
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item The substitution $x=\cos\theta$ turns trigonometric identities into polynomial recurrences.
\item The Chebyshev polynomials of the first kind are defined by
\[
T_n(x)=\cos(n\theta)\qquad (x=\cos\theta).
\]
\item The Chebyshev polynomials of the second kind are defined by
\[
U_n(x)=\frac{\sin((n+1)\theta)}{\sin\theta}\qquad (x=\cos\theta).
\]
\item These recurrences give fast formulas for $\cos(n\theta)$ without repeated expansion.
\item \textit{Roots are Predictable}: Because these polynomials are born from a cosine wave, their roots will always be real, distinct, and trapped strictly between $-1$ and $1$. In higher mathematics, these roots (called "Chebyshev Nodes") are the optimal points to pick when trying to curve-fit data!
\item \textit{Chebyshev Polynomials are Everywhere}: These polynomials are not just a theoretical curiosity; they play a crucial role in numerical analysis, approximation theory, and even in your 5G networks. 
\end{itemize}
\end{takeaways}

\begin{problem}[Hermite and the Bell Curve]

    % The Gaussian function
% \[
% f(x)=e^{-x^2}
% \]
% generates the physicist's Hermite polynomials
% \[
% H_n(x)=(-1)^n e^{x^2}\frac{d^n}{dx^n}\left(e^{-x^2}\right),\qquad n\ge 0.
% \]


The Gaussian function, $f(x) = e^{-x^2}$, is the mathematical foundation of the famous "Bell Curve" used in probability and statistics. By taking successive derivatives of this curve, we generate a family of functions known as the "Physicist's Hermite Polynomials" denoted as $H_n(x)$.They are defined explicitly by Rodrigues' Formula:

\[ H_n(x) = (-1)^n e^{x^2} \frac{d^n}{dx^n} \left( e^{-x^2} \right) \quad \text{for } n \ge 0. \]

(Note: $\frac{d^0}{dx^0}$ implies no differentiation, so $H_0(x) = 1$)


\begin{enumerate}[label=(\roman*)]
\item By manually applying the Chain Rule and Product Rule, find explicit formulas for $H_1(x)$, $H_2(x)$, and $H_3(x)$.

\item Find the roots of $H_3(x)=0$. Hence state a conjecture about the parity of $H_n(x)$ when $n$ is even or odd.

\item Starting from
\[
(-1)^n\frac{d^n}{dx^n}\left(e^{-x^2}\right)=e^{-x^2}H_n(x),
\]
prove that
\[
H_{n+1}(x)=2xH_n(x)-H_n'(x).
\]
\item \textbf{Quantum Mechanics Application:} In quantum physics, the probability of finding a vibrating particle at position $x$ in its $n$th energy state is proportional to
\[
P_n(x)=e^{-x^2}[H_n(x)]^2.
\]
For the second excited state ($n=2$), use your formula for $H_2(x)$ to build $P_2(x)$, find $P_2'(x)$, and determine the exact $x$-coordinates of the local maxima.
\end{enumerate}
\end{problem}

\begin{hintbox}
Differentiate step by step in part (i). In part (iii), differentiate both sides and use the product rule carefully. Part (iv) is left as an extension challenge for readers curious about quantum mechanics.
\end{hintbox}

\begin{solution}
\textbf{(i)} First,
\[
H_1(x)=-e^{x^2}\frac{d}{dx}(e^{-x^2})
=-e^{x^2}(-2xe^{-x^2})
=\boxed{2x}.
\]

Next,
\[
\frac{d^2}{dx^2}(e^{-x^2})
=\frac{d}{dx}(-2xe^{-x^2})
=-2e^{-x^2}+4x^2e^{-x^2}.
\]
So
\[
H_2(x)=e^{x^2}e^{-x^2}(4x^2-2)=\boxed{4x^2-2}.
\]

Then
\[
\frac{d^3}{dx^3}(e^{-x^2})
=\frac{d}{dx}\bigl(e^{-x^2}(4x^2-2)\bigr)
=e^{-x^2}(-8x^3+12x).
\]
Hence
\[
H_3(x)=-e^{x^2}e^{-x^2}(-8x^3+12x)=\boxed{8x^3-12x}.
\]

\textbf{(ii)} Solve
\[
8x^3-12x=0
\Rightarrow 4x(2x^2-3)=0.
\]
So the roots are
\[
\boxed{x=0,\ \pm\sqrt{\frac32}.}
\]
Also $H_1$ and $H_3$ are odd, while $H_0$ and $H_2$ are even. So the natural conjecture is:
\[
\boxed{H_n(x)\text{ is even if }n\text{ is even, and odd if }n\text{ is odd}.}
\]

\textbf{(iii)} Differentiate
\[
(-1)^n\frac{d^n}{dx^n}(e^{-x^2})=e^{-x^2}H_n(x).
\]
The left-hand side becomes
\[
(-1)^n\frac{d^{n+1}}{dx^{n+1}}(e^{-x^2})
=-e^{-x^2}H_{n+1}(x).
\]
The right-hand side gives
\[
\frac{d}{dx}\bigl(e^{-x^2}H_n(x)\bigr)
=(-2xe^{-x^2})H_n(x)+e^{-x^2}H_n'(x).
\]
So
\[
-e^{-x^2}H_{n+1}(x)=e^{-x^2}\bigl(-2xH_n(x)+H_n'(x)\bigr).
\]
Divide by $-e^{-x^2}$:
\[
\boxed{H_{n+1}(x)=2xH_n(x)-H_n'(x).}
\]

\textbf{(iv)} Left as a challenge problem. This is a lovely invitation to explore quantum mechanics, where Hermite polynomials describe the quantum harmonic oscillator.
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item A good recurrence can replace repeated heavy differentiation.
\item Hermite polynomials alternate parity: even index gives an even polynomial, odd index gives an odd polynomial.
\item Rodrigues' formula is a powerful bridge between calculus and polynomial structure.
\item Hermite polynomials also appear naturally in quantum mechanics.
\end{itemize}
\end{takeaways}

\begin{problem}[Bernoulli Polynomials and the Beta Distribution]

    % The Bernoulli polynomials start with $B_0(x)=1$ and satisfy, for $n\ge 1$:
% \[
% \frac{d}{dx}B_n(x)=nB_{n-1}(x),
% \qquad
% \int_0^1 B_n(x)\,dx=0.
% \]
% Given
% \[
% B_2(x)=x^2-x+\frac16,
% \]
% answer the following:

The Bernoulli polynomials, $B_n(x)$, were discovered by Swiss mathematician Jacob Bernoulli in the early 1700s (published posthumously in his masterwork Ars Conjectandi). Originally developed to solve centuries-old number theory puzzles, these polynomials have become surprisingly crucial in modern applied mathematics. In higher education, they form the foundation of the Euler-Maclaurin formula (used in numerical analysis to approximate complex integrals) and frequently appear in advanced statistics to define the moments, cumulants, and error bounds of Probability Density Functions (PDFs).

They are defined by a starting seed $B_0(x) = 1$, and two strict calculus rules for all integers $n \ge 1$:

(1) The Derivative Rule: $\frac{d}{dx} B_n(x) = n B_{n-1}(x)$
(2) The Integral Rule: $\int_0^1 B_n(x) \, dx = 0$

Given that $B_2(x) = x^2 - x + \frac{1}{6}$

\begin{enumerate}[label=(\roman*)]
\item Use the Derivative and Integral rules to find the explicit algebraic expressions for $B_3(x)$ and $B_4(x)$.
\item A remarkable property of Bernoulli polynomials is their symmetry. Verify algebraically that $B_3(1-x) = -B_3(x)$. Hence, without further calculation, state the three roots of the cubic equation $B_3(x) = 0$.
\item In probability theory, a valid Probability Density Function (PDF) defined on the interval $0 \le x \le 1$ must satisfy two conditions: $f(x) \ge 0$ for all $x$, and $\int_0^1 f(x) \, dx = 1$.
Consider the function $f(x) = k \left( B_4(x) + \frac{1}{30} \right)$.
Use the Integral Rule to quickly determine the value of the constant $k$. Then, use your polynomial $B_4(x)$ from part (i) to factorise $f(x)$ and explicitly justify why $f(x) \ge 0$ on this domain.
% Let
% \[
% f(x)=k\left(B_4(x)+\frac{1}{30}\right).
% \]
% Find $k$, factorise $f(x)$, and justify that $f(x)\ge 0$ on $0\le x\le 1$.
\item The distribution defined in part (iii) is a special case of the Beta Distribution, a continuous probability distribution widely used in higher-level statistics to model random proportions (such as the reliability of a mechanical system or the conversion rate of a website).
Using your factorised form of $f(x)$, calculate the Expected Value, $E(X)$, of this distribution.

% Find the expected value
% \[
% E(X)=\int_0^1 x f(x)\,dx.
% \]


\end{enumerate}
\end{problem}

\begin{hintbox}
Integrate to get $B_3$ and $B_4$, then use the condition $\int_0^1 B_n(x)\,dx=0$ to find the constants. In part (iii), do not integrate $B_4$ directly: use the rule you were given.
\end{hintbox}

\begin{solution}
\textbf{(i)} Since
\[
B_3'(x)=3B_2(x)=3x^2-3x+\frac12,
\]
integration gives
\[
B_3(x)=x^3-\frac32x^2+\frac12x+C.
\]
Using $\int_0^1 B_3(x)\,dx=0$ gives $C=0$, so
\[
\boxed{B_3(x)=x^3-\frac32x^2+\frac12x.}
\]

Next,
\[
B_4'(x)=4B_3(x)=4x^3-6x^2+2x,
\]
so
\[
B_4(x)=x^4-2x^3+x^2+D.
\]
Using $\int_0^1 B_4(x)\,dx=0$ gives $D=-\frac1{30}$, hence
\[
\boxed{B_4(x)=x^4-2x^3+x^2-\frac1{30}.}
\]

\textbf{(ii)} Substitute $1-x$ into $B_3$:
\[
B_3(1-x)=-(x^3-\tfrac32x^2+\tfrac12x)=-B_3(x).
\]
Also
\[
B_3(x)=x\left(x^2-\frac32x+\frac12\right)=x(x-1)\left(x-\frac12\right),
\]
so the roots are
\[
\boxed{0,\ \frac12,\ 1.}
\]

\textbf{(iii)} Since $\int_0^1 f(x)\,dx=1$,
\[
k\int_0^1\left(B_4(x)+\frac1{30}\right)dx=1.
\]
Using $\int_0^1 B_4(x)\,dx=0$,
\[
\frac{k}{30}=1 \Rightarrow \boxed{k=30.}
\]
Now
\[
f(x)=30\left(x^4-2x^3+x^2\right)=30x^2(x-1)^2.
\]
This is non-negative on $[0,1]$, since it is a product of squares.

\textbf{(iv)} 
\[
E(X)=\int_0^1 x f(x)\,dx
=30\int_0^1 x^3(x-1)^2\,dx.
\]
So
\[
E(X)=30\int_0^1 (x^5-2x^4+x^3)\,dx
=30\left(\frac16-\frac25+\frac14\right)
=\boxed{\frac12.}
\]
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item The derivative and zero-integral rules determine Bernoulli polynomials recursively.
\item Symmetry can reveal roots quickly.
\item \textit{Beta Distributions} is a family of continuous probability distributions defined on the interval $[0, 1]$. They are parameterized by two positive shape parameters, $\alpha$ and $\beta$, which determine the shape of the distribution. The Beta distribution is widely used in statistics to model random variables that represent proportions or probabilities, such as the success rate of a process or the proportion of a population with a certain characteristic. It is defined by the probability density function:
\[f(x; \alpha, \beta) = \frac{x^{\alpha - 1}(1 - x)^{\beta - 1}}{B(\alpha, \beta)}\]
where $B(\alpha, \beta)$ is the Beta function, which serves as a normalization constant to ensure that the total area under the curve is 1 and is given by:
\[B(\alpha, \beta) = \int_0^1 t^{\alpha - 1}(1 - t)^{\beta - 1} dt.\]
\item \textit{Beta Distributions in the Real World}: The function $f(x) = 30x^2(1-x)^2$ is formally known as a $\text{Beta}(3,3)$ distribution. Because its domain is strictly $[0,1]$, university statisticians use it to model percentages and probabilities, such as predicting the batting average of a baseball player or the defect rate in a manufacturing plant!
\item \textit{Interconnected Math}: This problem beautifully demonstrates that pure calculus (polynomial derivation) and applied statistics (Probability Density Functions) are not isolated topics. The algebraic properties of a polynomial directly dictate its viability as a statistical model.

\end{itemize}
\end{takeaways}

\begin{problem}[Legendre Polynomials and the Infinite Dot Product]


% For functions on $[-1,1]$, define the inner product by
% \[
% \langle f,g\rangle=\int_{-1}^{1}f(x)g(x)\,dx.
% \]
% Two functions are orthogonal if this integral is $0$.

% The first three Legendre polynomials are
% \[
% P_0(x)=1,\qquad P_1(x)=x,\qquad P_2(x)=\frac12(3x^2-1).
% \]


In the study of vectors, two vectors are defined as orthogonal (perpendicular) if their dot product is zero: $\vec{u} \cdot \vec{v} = 0$.

In advanced calculus, we can apply this exact same concept to functions. We define the "inner product" of two functions $f(x)$ and $g(x)$ over the interval $[-1, 1]$ as the definite integral of their product. Thus, two functions are \textbf{orthogonal} if:
\[
\int_{-1}^{1} f(x)g(x) \, dx = 0.
\]
The \textbf{Legendre Polynomials}, denoted $P_n(x)$, form a sequence of orthogonal polynomials. The first three are:
\[
P_0(x) = 1,\quad P_1(x) = x,\quad P_2(x) = \frac{1}{2}(3x^2 - 1).
\]

\begin{enumerate}[label=(\roman*)]
\item Verify that $P_0$ and $P_1$ are orthogonal, and that $P_1$ and $P_2$ are orthogonal.
\item Let $Q(x)=ax^2+bx+c$. Show that if $Q$ is orthogonal to both $P_0$ and $P_1$, then $Q$ is a scalar multiple of $P_2$.
\item Using
\[
(n+1)P_{n+1}(x)=(2n+1)xP_n(x)-nP_{n-1}(x),
\]
(do NOT prove this) find $P_3(x)$.
\item \textbf{Application to Numerical Analysis:} 
% The two-point Gaussian Quadrature rule estimates

 In computer science, numerical integration usually provides an estimate rather than a perfect answer. The two-point Gaussian Quadrature rule estimates the integral of a general function $g(x)$ by sampling it at the roots of the Legendre polynomial $P_2(x)$:

\[
\int_{-1}^{1} g(x)\,dx \approx g(\alpha)+g(\beta),
\]
where $\alpha$ and $\beta$ are the roots of $P_2(x)=0$.

Remarkably, the theorem guarantees that this is not just an approximation, but an exact equality for any polynomial of degree 3 or less.

Verify that for every cubic
\[
f(x)=ax^3+bx^2+cx+d,
\]
the formula is exact:
\[
\int_{-1}^{1} f(x)\,dx=f(\alpha)+f(\beta).
\]
\end{enumerate}
\end{problem}

\begin{hintbox}
For part (ii), write down the two orthogonality integrals and simplify. For part (iv), first solve $P_2(x)=0$, then use the symmetry $\beta=-\alpha$ to simplify the odd-power terms.
\end{hintbox}

\begin{solution}
\textbf{(i)} 
\[
\int_{-1}^{1}P_0(x)P_1(x)\,dx=\int_{-1}^{1}x\,dx=0.
\]
Also
\[
\int_{-1}^{1}P_1(x)P_2(x)\,dx
=\frac12\int_{-1}^{1}(3x^3-x)\,dx=0,
\]
since the integrand is odd.

\textbf{(ii)} Let $Q(x)=ax^2+bx+c$.
Orthogonal to $P_0$:
\[
\int_{-1}^{1}(ax^2+bx+c)\,dx=\frac{2a}{3}+2c=0
\Rightarrow c=-\frac{a}{3}.
\]
Orthogonal to $P_1$:
\[
\int_{-1}^{1}(ax^2+bx+c)x\,dx=\int_{-1}^{1}(ax^3+bx^2+cx)\,dx=\frac{2b}{3}=0,
\]
so $b=0$. Hence
\[
Q(x)=ax^2-\frac{a}{3}=\frac{a}{3}(3x^2-1)=\frac{2a}{3}P_2(x).
\]
So $Q$ is a scalar multiple of $P_2$.

\textbf{(iii)} Put $n=2$:
\[
3P_3(x)=5xP_2(x)-2P_1(x).
\]
Thus
\[
3P_3(x)=5x\cdot \frac12(3x^2-1)-2x=\frac{15x^3-9x}{2},
\]
so
\[
\boxed{P_3(x)=\frac12(5x^3-3x).}
\]

\textbf{(iv)} The roots of $P_2(x)=\frac12(3x^2-1)$ are 
$\alpha=\frac{1}{\sqrt3},\qquad \beta=-\frac{1}{\sqrt3}.$
Now
\[
\int_{-1}^{1}(ax^3+bx^2+cx+d)\,dx=\frac{2b}{3}+2d,
\]
while
\[
f(\alpha)+f(\beta)
=a(\alpha^3+\beta^3)+b(\alpha^2+\beta^2)+c(\alpha+\beta)+2d
=\frac{2b}{3}+2d,
\]
because $\beta=-\alpha$, so the odd-power terms cancel and
\[
\alpha^2+\beta^2=\frac13+\frac13=\frac23.
\]
Hence
\[
\boxed{\int_{-1}^{1} f(x)\,dx=f(\alpha)+f(\beta).}
\]
This is a beautiful example of Gaussian quadrature. The general theory needs more heavy lifting and quite a bit of manual calculation.
\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
\item Orthogonality of functions is the continuous version of a dot product being zero.
\item Symmetry on $[-1,1]$ is a powerful shortcut: odd integrands integrate to zero.
\item Legendre, Chebyshev, and Hermite polynomials are all examples of orthogonal polynomial families.
\item The roots of orthogonal polynomials also power Gaussian quadrature, one of the most important ideas in numerical integration.
\item \textit{Calculus Meets Vectors}: In the HSC syllabus, you learn the dot product as $\vec{u} \cdot \vec{v} = u_1v_1 + u_2v_2 + u_3v_3 = 0$. An integral is simply an infinite sum of tiny slices. Therefore, $\int f(x)g(x) \, dx = 0$ is literally the dot product of two continuous functions with infinitely many dimensions!
\end{itemize}
\end{takeaways}
