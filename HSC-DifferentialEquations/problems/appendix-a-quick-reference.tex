\begin{itemize}[leftmargin=*]
  \item \textbf{General first-order form:} $\dfrac{dy}{dx} = F(x,y)$.
  \item \textbf{Initial condition:} a value such as $y(x_0)=y_0$ used to pick out one particular solution.
  \item \textbf{Separable forms:}
  \[
  \frac{dy}{dx}=f(x), \qquad
  \frac{dy}{dx}=g(y), \qquad
  \frac{dy}{dx}=f(x)g(y).
  \]
  \item \textbf{Core integration idea:} if variables can be separated, rewrite the equation in the form
  \[
  A(y)\,dy = B(x)\,dx
  \]
  and integrate both sides.
  \item \textbf{Equilibrium solution:} a constant solution $y=k$ for which $\dfrac{dy}{dx}=0$.
  \item \textbf{Slope field reading rule:} each small segment shows the local tangent direction of a solution curve.
  \item \textbf{Exponential-style model:}
  \[
  \frac{dy}{dx}=ky
  \qquad \Longrightarrow \qquad
  y=Ae^{kx}.
  \]
  \item \textbf{Logistic model:}
  \[
  \frac{dy}{dx}=ry\left(1-\frac{y}{M}\right),
  \]
  where $M$ is the carrying capacity and the equilibrium values are $y=0$ and $y=M$.
  \item \textbf{Newton's law of cooling:}
  \[
  \frac{dT}{dt}=-k(T-T_a)
  \qquad \Longrightarrow \qquad
  T(t)=T_a+Ce^{-kt}.
  \]
  \item \textbf{Torricelli-style draining model:}
  \[
  \frac{dh}{dt}=-k\sqrt{h}
  \qquad \Longrightarrow \qquad
  2\sqrt{h}=-kt+C.
  \]
  \item \textbf{Particle moving on a curve:}
  \[
  \mathbf{r}(t)=x(t)\mathbf{i}+y(t)\mathbf{j}, \qquad
  \mathbf{v}(t)=\dot{x}\mathbf{i}+\dot{y}\mathbf{j}, \qquad
  \mathbf{a}(t)=\ddot{x}\mathbf{i}+\ddot{y}\mathbf{j}.
  \]
  % \item \textbf{Interpretation checklist:} identify units, define constants clearly, apply the initial condition, and state what the long-run behavior means in context.
\end{itemize}
