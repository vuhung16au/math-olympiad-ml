\begin{itemize}[leftmargin=*]
  \item Arithmetic nth term:
  \[
a_n=a+(n-1)d.
\]
  \item Arithmetic sum of first $n$ terms:
  \[
S_n=\frac{n}{2}(2a+(n-1)d)=\frac{n}{2}(a+l).
\]
  \item Geometric nth term:
  \[
a_n=ar^{n-1}.
\]
  \item Geometric sum of first $n$ terms ($r\ne1$):
  \[
S_n=a\frac{1-r^n}{1-r}=a\frac{r^n-1}{r-1}.
\]
  \item Infinite geometric sum ($|r|<1$):
  \[
S_\infty=\frac{a}{1-r}.
\]
  \item Sums of powers for the first $n$ positive integers:
  \[
\sum_{k=1}^{n}k=\frac{n(n+1)}{2},
\]
\[
\sum_{k=1}^{n}k^2=\frac{n(n+1)(2n+1)}{6},
\]
\[
\sum_{k=1}^{n}k^3=\left(\frac{n(n+1)}{2}\right)^2.
\]
\end{itemize}

\subsection*{Chebyshev Polynomials}
The Chebyshev polynomials of the first kind, denoted $T_n(x)$, are defined by
\[
T_0(x)=1,\qquad T_1(x)=x,\qquad
T_{n+1}(x)=2xT_n(x)-T_{n-1}(x)\quad (n\ge 1).
\]
Equivalently,
\[
T_n(\cos\theta)=\cos(n\theta).
\]

The Chebyshev polynomials of the second kind, denoted $U_n(x)$, are defined by
\[
U_0(x)=1,\qquad U_1(x)=2x,\qquad
U_{n+1}(x)=2xU_n(x)-U_{n-1}(x)\quad (n\ge 1).
\]
Equivalently,
\[
U_n(\cos\theta)=\frac{\sin((n+1)\theta)}{\sin\theta}\quad (\sin\theta\ne 0).
\]
