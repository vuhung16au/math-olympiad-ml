\begin{problem}[Quick AP term]
Find $a_{18}$ for the arithmetic sequence $2,7,12,\ldots$.
\end{problem}
\begin{hint}
Use $a_n=a+(n-1)d$ with $a=2$ and $d=5$.
\end{hint}
\begin{solution}
\[
a_{18}=2+17\cdot 5=87.
\]
\end{solution}

\begin{problem}[Quick GP sum]
Find
\[
1+\frac12+\frac14+\frac18+\cdots+\frac1{128}.
\]
\end{problem}
\begin{hint}
This is a finite GP with $a=1$, $r=\frac12$, and $8$ terms.
\end{hint}
\begin{solution}
\[
S_8=\frac{1-(\frac12)^8}{1-\frac12}=2\left(1-\frac1{256}\right)=\frac{255}{128}.
\]
\end{solution}

\begin{problem}[Quick Chebyshev value]
For the second-kind Chebyshev polynomial
\[
U_2(x)=4x^2-1,
\]
find $U_2\left(\frac12\right)$.
\end{problem}
\begin{hint}
Substitute $x=\frac12$ carefully.
\end{hint}
\begin{solution}
\[
U_2\left(\frac12\right)=4\left(\frac14\right)-1=1-1=0.
\]
\end{solution}
\begin{takeaways}
\item For a given Chebyshev polynomial, direct substitution is often enough.
\item Square fractions before multiplying by the coefficient.
\end{takeaways}

\begin{problem}[Arithmetic pricing model]
An installer charges \$400 for the first window and \$100 for each additional window.
\begin{enumerate}[(i)]
\item Write the total cost for 1, 2, 3, and 4 windows.
\item Find a formula for the total cost $A_n$ for $n$ windows.
\item Find $A_{15}$.
\item If the budget is strictly less than \$10,000, what is the maximum number of windows?
\end{enumerate}
\end{problem}
\begin{hint}
Model $A_n$ as an arithmetic sequence with first term $400$ and difference $100$.
\end{hint}
\begin{solution}
\[
A_n=400+(n-1)100=100n+300.
\]
So $A_{15}=1800$. For the budget:
\[
100n+300<10000 \implies n<97,
\]
hence the maximum integer is $n=96$.
\end{solution}

\begin{problem}[Simple interest and AP]
A principal of \$2,000,000 is invested at $5.7\%$ p.a. simple interest.
Let $A_n$ be the total amount after $n$ years.
\begin{enumerate}[(i)]
\item Find $A_1,A_2,A_3,A_4$.
\item Find a formula for $A_n$ and evaluate $A_{12}$.
\item How many whole years before the amount exceeds \$6,000,000?
\end{enumerate}
\end{problem}
\begin{hint}
Simple interest adds a fixed amount each year, so $(A_n)$ is arithmetic.
\end{hint}
\begin{solution}
Yearly interest is $0.057\times 2{,}000{,}000=114{,}000$.
Hence
\[
A_n=2{,}000{,}000+114{,}000n.
\]
So
\[
A_1=2{,}114{,}000,\ A_2=2{,}228{,}000,\ A_3=2{,}342{,}000,\ A_4=2{,}456{,}000,
\]
and
\[
A_{12}=3{,}368{,}000.
\]
For exceeding \$6,000,000:
\[
2{,}000{,}000+114{,}000n>6{,}000{,}000
\implies n>35.087\ldots
\]
so $n=36$ years.
\end{solution}

\begin{problem}[Recurring decimal to fraction]
Convert each to a fraction:
\[
0.2,\qquad 0.\dot{2},\qquad 0.1\dot{2}.
\]
\end{problem}
\begin{hint}
Use place value for terminating decimals and GP ideas for recurring decimals.
\end{hint}
\begin{solution}
\[
0.2=\frac15,\qquad 0.\dot{2}=\frac29,\qquad 0.1\dot{2}=\frac{11}{90}.
\]
\end{solution}

\begin{problem}[Nested shaded squares]
In a unit square, shade the bottom-left quarter. Then repeatedly subdivide the top-right quarter into four equal squares and shade its bottom-left quarter.
Find the total shaded fraction.
\end{problem}
\begin{hint}
Shaded areas form an infinite GP with first term $\frac14$ and ratio $\frac14$.
\end{hint}
\begin{solution}
\[
S_\infty=\frac{\frac14}{1-\frac14}=\frac13.
\]
So the shaded fraction is $\frac13$.
\end{solution}

\begin{problem}[Fraction of a recurring decimal]
Find the fraction for $0.\dot{1}\dot{2}\dot{3}$?
\end{problem}
\begin{hint}
Use $a = \frac{123}{1000}$ and $r = \frac{1}{1000}$ in your $S_\infty$ formula.
\end{hint}
\begin{solution}
\[
S_\infty = \frac{\frac{123}{1000}}{1 - \frac{1}{1000}} = \frac{123}{999} = \frac{41}{333}.
\]
\end{solution}
\begin{takeaways}
\item Every recurring decimal is a geometric series sum $S_\infty$.
\end{takeaways}
