\begin{problem}[Quick AP term]
Find $a_{18}$ for the arithmetic sequence $2,7,12,\ldots$.
\end{problem}
\begin{hint}
Use $a_n=a+(n-1)d$ with $a=2$ and $d=5$.
\end{hint}
\begin{solution}
\[
a_{18}=2+17\cdot 5=87.
\]
\end{solution}

\begin{problem}[Quick GP sum]
Find
\[
1+\frac12+\frac14+\frac18+\cdots+\frac1{128}.
\]
\end{problem}
\begin{hint}
This is a finite GP with $a=1$, $r=\frac12$, and $8$ terms.
\end{hint}
\begin{solution}
\[
S_8=\frac{1-(\frac12)^8}{1-\frac12}=2\left(1-\frac1{256}\right)=\frac{255}{128}.
\]
\end{solution}
