\begin{problem}[Arithmetic nth term]
An arithmetic sequence has first term $4$ and common difference $5$.
Find $a_{30}$.
\end{problem}
\begin{solution}
Use $a_n=a+(n-1)d$:
\[
a_{30}=4+29\cdot 5=149.
\]
\end{solution}

\begin{problem}[Find the common ratio]
In a geometric sequence, $a_2=12$ and $a_5=324$. Find the common ratio $r$ and the first term $a_1$.
\end{problem}
\begin{solution}
From $a_n=ar^{n-1}$,
\[
a_2=ar=12,\qquad a_5=ar^4=324.
\]
Divide to remove $a$:
\[
\frac{a_5}{a_2}=r^3=\frac{324}{12}=27 \implies r=3.
\]
Then $a_1=a=\frac{12}{3}=4$.
\end{solution}

\begin{problem}[Sum of an AP]
Find the sum of the first $40$ terms of the arithmetic sequence
\[
9,\ 13,\ 17,\ \ldots
\]
\end{problem}
\begin{solution}
Here $a=9$, $d=4$, $n=40$.
The 40th term is
\[
l=9+39\cdot 4=165.
\]
So
\[
S_{40}=\frac{40}{2}(9+165)=20\cdot 174=3480.
\]
\end{solution}

\begin{problem}[Recurring decimal as a fraction]
Express $0.\overline{23}$ as a rational number.
\end{problem}
\begin{solution}
\[
0.\overline{23}=\frac{23}{99}.
\]
A geometric view is
\[
0.23+0.0023+0.000023+\cdots
\]
with first term $\frac{23}{100}$ and ratio $\frac{1}{100}$.
\end{solution}
