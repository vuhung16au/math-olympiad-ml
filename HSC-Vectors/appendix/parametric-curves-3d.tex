% Row 1
\noindent
\begin{minipage}[t]{0.48\textwidth}
\centering
\tdplotsetmaincoords{70}{120}
\begin{tikzpicture}[tdplot_main_coords, scale=0.42]
  \draw[thin,->] (0,0,0) -- (1.6,0,0);
  \draw[thin,->] (0,0,0) -- (0,1.6,0);
  \draw[thin,->] (0,0,0) -- (0,0,3.8);
  \draw[thick,domain=0:12.566,samples=100,smooth]
    plot ({cos(\x r)},{sin(\x r)},{\x/3.14159});
\end{tikzpicture}\\[-0.1em]
\textbf{The Helix (Slinky)}\\[-0.1em]
\scriptsize Simplest continuous spiral in three dimensions.\\[0.1em]
$x=\cos t$, \quad $y=\sin t$, \quad $z=t$
\end{minipage}\hfill
\begin{minipage}[t]{0.48\textwidth}
\centering
\tdplotsetmaincoords{70}{120}
\begin{tikzpicture}[tdplot_main_coords, scale=0.22]
  \draw[thin,->] (0,0,0) -- (3.2,0,0);
  \draw[thin,->] (0,0,0) -- (0,3.2,0);
  \draw[thin,->] (0,0,0) -- (0,0,1.2);
  \draw[thick,domain=0:360,samples=100,smooth]
    plot ({cos(\x)+2*cos(2*\x)},{sin(\x)-2*sin(2*\x)},{-sin(3*\x)});
\end{tikzpicture}\\[-0.1em]
\textbf{Torus Knot (Figure-Eight)}\\[-0.1em]
\scriptsize A tube wraps along a closed, twisting figure-eight path.\\[0.1em]
$x=\cos t+2\cos 2t$\\
$y=\sin t-2\sin 2t$, \quad $z=-\sin 3t$
\end{minipage}

\textbf{3D Rose (Rhodonea)}\\[-0.1em]
\scriptsize A 2D flower curve lifted into 3D by a periodic $z$-component.\\[0.1em]
$x=\cos(4t)\cos t$, \quad $y=\cos(4t)\sin t$, \quad $z=\sin(4t)$
\end{minipage}\hfill
\begin{minipage}[t]{0.48\textwidth}
\centering

  \draw[thin,->] (0,0,0) -- (0,0,1.2);
  \draw[thick,domain=0:360,samples=100,smooth]
    plot ({sin(3*\x)},{sin(4*\x)},{sin(\x)});
\end{tikzpicture}\\[-0.1em]
\textbf{3D Lissajous Curve}\\[-0.1em]
\scriptsize Wavelike motion on all three axes creates symmetric nodes.\\[0.1em]
$x=\sin(3t)$, \quad $y=\sin(4t)$, \quad $z=\sin t$
\end{minipage}

    plot ({4*sin(\x)^3},{0.25*(13*cos(\x)-5*cos(2*\x)-2*cos(3*\x)-cos(4*\x))},0);
\end{tikzpicture}\\[-0.1em]
\textbf{Heart}\\[-0.1em]
\scriptsize Classic heart profile in the $xy$-plane ($z=0$).\\[0.1em]
$x=4R\sin^3 u$\\
$y=\tfrac{R}{4}\bigl(13\cos u-5\cos 2u-2\cos 3u-\cos 4u\bigr)$
\end{minipage}\hfill
\begin{minipage}[t]{0.48\textwidth}
\centering

    plot ({cos(\x r)},{sin(\x r)},{cos(\x r)});
\end{tikzpicture}\\[-0.1em]
\textbf{Spring}\\[-0.1em]
\scriptsize Circular motion in $xy$ with oscillating height in $z$.\\[0.1em]
$x=R\cos t$, \quad $y=R\sin t$, \quad $z=R\cos t$
\end{minipage}

  \draw[thick,domain=0:360,samples=80,smooth]
    plot ({2*cos(\x)+0.35*cos(\x/2)*cos(\x)},{2*sin(\x)+0.35*cos(\x/2)*sin(\x)},{0.35*sin(\x/2)});
\end{tikzpicture}\\[-0.1em]
\textbf{M\"obius Strip}\\[-0.1em]
\scriptsize Non-orientable surface; centre circle ($v=0$) and edge ($v=1$) shown.\\[0.1em]
$x=2R\bigl(\cos u+v\cos\tfrac{u}{2}\cos u\bigr)$\\
$y=2R\bigl(\sin u+v\cos\tfrac{u}{2}\sin u\bigr)$, \quad $z=2Rv\sin\tfrac{u}{2}$
\end{minipage}\hfill

\textbf{Cone}\\[-0.1em]
\scriptsize Spiral path on a conical surface as radius and height grow.\\[0.1em]
$x=t\cos t$, \quad $y=t\sin t$, \quad $z=t$
\end{minipage}
