\begin{problem}[Expanding a binomial surd expression]
Expand
\[
(1-\sqrt{2})^5=a+b\sqrt{2},
\]
where $a$ and $b$ are integers. Find $a$ and $b$.
\end{problem}

\begin{hint}
Use the binomial theorem, then simplify powers of $\sqrt{2}$ carefully: $(\sqrt{2})^2=2$, $(\sqrt{2})^3=2\sqrt{2}$, and so on.
\end{hint}

\begin{solution}
Using the binomial theorem,
\begin{align*}
(1-\sqrt{2})^5
&= 1 - 5\sqrt{2} + 10(\sqrt{2})^2 - 10(\sqrt{2})^3 + 5(\sqrt{2})^4 - (\sqrt{2})^5 \\
&= 1 - 5\sqrt{2} + 20 - 20\sqrt{2} + 20 - 4\sqrt{2} \\
&= 41 - 29\sqrt{2}.
\end{align*}
Hence
\[
\boxed{a=41,\quad b=-29}.
\]
\end{solution}

\begin{takeaways}
\item After expanding, separate the rational terms from the $\sqrt{2}$ terms.
\item Powers of $\sqrt{2}$ alternate between integer multiples and surd multiples.
\end{takeaways}

\begin{problem}[When is a conjugate surd sum rational?]
For what integer values of $n$ is
\[
(1-\sqrt{2})^n+(1+\sqrt{2})^n
\]
rational?
\end{problem}

\begin{hint}
Expand both powers using the binomial theorem and compare the terms involving odd and even powers of $\sqrt{2}$. Also think about what happens for negative integers.
\end{hint}

\begin{solution}
For a non-negative integer $n$, expand both terms by the binomial theorem:
\[
(1+\sqrt{2})^n=\binom{n}{0}+\binom{n}{1}\sqrt{2}+\binom{n}{2}(\sqrt{2})^2+\cdots,
\]
\[
(1-\sqrt{2})^n=\binom{n}{0}-\binom{n}{1}\sqrt{2}+\binom{n}{2}(\sqrt{2})^2-\cdots.
\]

When we add these two expansions, all terms involving odd powers of $\sqrt{2}$ cancel, while the even-power terms remain. Since
\[
(\sqrt{2})^{2k}=2^k,
\]
all remaining terms are rational.

Therefore, for every non-negative integer $n$,
\[
(1-\sqrt{2})^n+(1+\sqrt{2})^n
\]
is rational.

Now consider negative integers. Since
\[
(1+\sqrt{2})(1-\sqrt{2})=1-2=-1,
\]
we have
\[
1-\sqrt{2}=-\frac{1}{1+\sqrt{2}}.
\]
So for any positive integer $n$,
\[
(1-\sqrt{2})^{-n}+(1+\sqrt{2})^{-n}
=(-1)^n(1+\sqrt{2})^n+(-1)^n(1-\sqrt{2})^n
=(-1)^n\Big((1+\sqrt{2})^n+(1-\sqrt{2})^n\Big),
\]
which is also rational.

Hence the expression is rational for
\[
\boxed{\text{all integers } n.}
\]
\end{solution}

\begin{takeaways}
\item Conjugate surds often become much simpler when added or multiplied.
\item In binomial expansions of $(a+b)^n$ and $(a-b)^n$, the odd-power terms cancel when the expansions are added.
\item Here that cancellation shows the expression is rational for every integer $n$.
\end{takeaways}

\begin{problem}[PINs with numbers and letters]
The number of possible different PINs with a combination of $4$ numbers and $2$ letters is:
\begin{enumerate}[label=\Alph*.]
\item $4435236$
\item $6760000$
\item $1000000$
\item $10676$
\end{enumerate}
\end{problem}

\begin{hint}
Use the multiplication principle: count the choices for the $4$ digits and the $2$ letters separately, then multiply.
\end{hint}

\begin{solution}
Assume the PIN has $4$ digits followed by $2$ letters, with repetition allowed.

There are $10$ choices for each digit and $26$ choices for each letter, so the total number of PINs is
\[
10^4 \times 26^2 = 10000 \times 676 = 6760000.
\]
So the correct answer is
\[
\boxed{\text{B. }6760000}.
\]
\end{solution}

\begin{takeaways}
\item If each position can be chosen independently, multiply the number of choices for each position.
\item Unless a question says otherwise, repetition is usually allowed in code or PIN questions.
\end{takeaways}

\begin{problem}[Two coins of the same metal]
A bag contains $n$ metal coins, where $n \ge 3$. There are $k$ silver coins in the bag and the rest are bronze.

Two coins are drawn at random from the bag, without replacement.

Find an expression for the probability that the two coins drawn are made of the same metal.
\end{problem}

\begin{hint}
The two coins are the same metal if they are both silver or both bronze.
\end{hint}

\begin{solution}
There are two favourable cases:

\textbf{Both silver:}
\[
\frac{k}{n}\cdot \frac{k-1}{n-1}.
\]

\textbf{Both bronze:}
There are $n-k$ bronze coins, so
\[
\frac{n-k}{n}\cdot \frac{n-k-1}{n-1}.
\]

Hence the required probability is
\[
\boxed{
\frac{k(k-1)}{n(n-1)}+\frac{(n-k)(n-k-1)}{n(n-1)}
}.
\]
\end{solution}

\begin{takeaways}
\item For ``same type'' probability questions, split into mutually exclusive cases and add.
\item Without replacement means the denominator changes on the second draw.
\end{takeaways}

\begin{problem}[Probabilities with the four jacks]
In a pack of cards, the $4$ jacks are taken out and shuffled.
\begin{enumerate}[label=\alph*)]
\item What is the probability of picking out the jack of diamonds at random?
\item If all the jacks are arranged in order, what is the probability of guessing the correct order?
\end{enumerate}
\end{problem}

\begin{hint}
For part (a), all four jacks are equally likely. For part (b), count the total number of possible orders of the four jacks.
\end{hint}

\begin{solution}
\textbf{a)} There are $4$ jacks, and only one of them is the jack of diamonds. So
\[
P(\text{jack of diamonds})=\frac{1}{4}.
\]

\textbf{b)} The $4$ jacks can be arranged in
\[
4!=24
\]
different orders. Only one of these is the correct order, so
\[
P(\text{guessing the correct order})=\frac{1}{24}.
\]
\end{solution}

\begin{takeaways}
\item If all outcomes are equally likely, probability is favourable outcomes over total outcomes.
\item Ordered arrangements of $4$ distinct objects are counted by $4!$.
\end{takeaways}

\begin{problem}[Counting rectangles in a $5\times 6$ grid]
How many rectangles can be found in the $5\times 6$ grid below? Remember that squares should also be counted, since every square is a rectangle.

\begin{center}
\begin{tikzpicture}[scale=0.75]
    \draw[step=1cm, black] (0,0) grid (6,5);
\end{tikzpicture}
\end{center}
\end{problem}

\begin{hint}
Choose any $2$ of the vertical grid lines and any $2$ of the horizontal grid lines.
\end{hint}

\begin{solution}
A rectangle is determined by choosing:
\begin{itemize}[leftmargin=*]
  \item $2$ of the $7$ vertical grid lines
  \item $2$ of the $6$ horizontal grid lines.
\end{itemize}

So the total number of rectangles is
\[
\binom{7}{2}\binom{6}{2}=21\times 15=315.
\]

Hence the number of rectangles is $\boxed{315}$.
\end{solution}

\begin{takeaways}
\item Grid-rectangle questions are usually solved by choosing boundary lines, not by listing shapes one by one.
\item Squares are already included, since every square is also a rectangle.
\end{takeaways}

\begin{problem}[How fast does brute force grow?]
A password is made from $n$ binary digits, each of which is either $0$ or $1$.

\begin{enumerate}[label=\alph*)]
\item How many possible passwords are there when $n=5$?
\item How many possible passwords are there when $n=10$?
\item Explain briefly why checking every possible password becomes difficult when $n$ is large.
\end{enumerate}
\end{problem}

\begin{hint}
Each position has $2$ choices, so use the multiplication principle.
\end{hint}

\begin{solution}
\textbf{a)} For $n=5$, the number of passwords is
\[
2^5=32.
\]

\textbf{b)} For $n=10$, the number of passwords is
\[
2^{10}=1024.
\]

\textbf{c)} The total number of possibilities doubles each time one more digit is added. So the number of passwords grows exponentially, and brute-force checking quickly becomes impractical for large $n$.
\end{solution}

\begin{takeaways}
\item Some counting problems grow exponentially, even when the rule looks simple.
\item This is one reason efficient counting methods matter.
\end{takeaways}

\begin{problem}[Repeated derivatives of $\frac{1}{x}$]
If
\[
f(x)=\frac{1}{x},
\]
find:
\begin{enumerate}[label=\roman*)]
\item $f'(x)$
\item $f''(x)$
\item $f^{(5)}(x)$
\item $f^{(n)}(x)$.
\end{enumerate}
\end{problem}

\begin{hint}
Write
\[
f(x)=x^{-1}
\]
and differentiate a few times to spot the pattern.
\end{hint}

\begin{solution}
Using the power rule,
\[
f(x)=x^{-1}.
\]

\textbf{i)}
\[
f'(x)=-x^{-2}=-\frac{1}{x^2}.
\]

\textbf{ii)}
\[
f''(x)=2x^{-3}=\frac{2}{x^3}.
\]

\textbf{iii)}
Continuing the pattern,
\[
f^{(3)}(x)=-\frac{6}{x^4},\qquad
f^{(4)}(x)=\frac{24}{x^5},
\]
so
\[
f^{(5)}(x)=-\frac{120}{x^6}.
\]

\textbf{iv)}
Each differentiation introduces:
\begin{itemize}[leftmargin=*]
  \item one extra factor in the numerator
  \item one extra power of $x$ in the denominator
  \item alternating signs.
\end{itemize}

Therefore
\[
\boxed{
f^{(n)}(x)=(-1)^n\frac{n!}{x^{n+1}}
}.
\]
\end{solution}

\begin{takeaways}
\item Repeated differentiation often reveals a pattern after the first few derivatives.
\item Here the coefficients are factorials: $1,1,2,6,24,120,\dots$.
\item This is more of a calculus enrichment problem than a core combinatorics problem, but it connects nicely to factorial notation.
\end{takeaways}

\begin{problem}[Using Pascal's triangle to differentiate $x^4$]
\begin{enumerate}[label=\roman*)]
\item Use Pascal's triangle to expand
\[
(x+h)^4.
\]

\item If
\[
f(x)=x^4,
\]
simplify
\[
f(x+h)-f(x).
\]

\item Hence use the definition
\[
f'(x)=\lim_{h\to 0}\frac{f(x+h)-f(x)}{h}
\]
to differentiate $x^4$.

\item For the general case
\[
f(x)=x^n,
\]
write down the expansion of $(x+h)^n$, and hence show that
\[
\frac{d}{dx}(x^n)=nx^{n-1}.
\]
\end{enumerate}
\end{problem}

\begin{hint}
Use row $4$ of Pascal's triangle for the specific case, then use the binomial theorem for $(x+h)^n$. In both cases, factor out $h$ before taking the limit.
\end{hint}

\begin{solution}
\textbf{i)} Using Pascal's triangle or the binomial theorem,
\[
(x+h)^4=x^4+4x^3h+6x^2h^2+4xh^3+h^4.
\]

\textbf{ii)} Since $f(x)=x^4$,
\[
f(x+h)-f(x)
=(x+h)^4-x^4
=4x^3h+6x^2h^2+4xh^3+h^4.
\]

\textbf{iii)} Therefore,
\begin{align*}
f'(x)
&=\lim_{h\to 0}\frac{f(x+h)-f(x)}{h}\\
&=\lim_{h\to 0}\frac{4x^3h+6x^2h^2+4xh^3+h^4}{h}\\
&=\lim_{h\to 0}\left(4x^3+6x^2h+4xh^2+h^3\right)\\
&=4x^3.
\end{align*}

Hence
\[
\boxed{f'(x)=4x^3.}
\]

\textbf{iv)} In general, the binomial theorem gives
\[
(x+h)^n=x^n+\binom{n}{1}x^{n-1}h+\binom{n}{2}x^{n-2}h^2+\cdots+h^n.
\]
So if
\[
f(x)=x^n,
\]
then
\[
f(x+h)-f(x)=nx^{n-1}h+\binom{n}{2}x^{n-2}h^2+\cdots+h^n.
\]
Therefore,
\begin{align*}
f'(x)
&=\lim_{h\to 0}\frac{f(x+h)-f(x)}{h}\\
&=\lim_{h\to 0}\left(nx^{n-1}+\binom{n}{2}x^{n-2}h+\cdots+h^{n-1}\right)\\
&=nx^{n-1}.
\end{align*}

Hence
\[
\boxed{\frac{d}{dx}(x^n)=nx^{n-1}.}
\]
\end{solution}

\begin{takeaways}
\item Pascal's triangle gives the coefficients in expansions such as $(x+h)^4$.
\item In the derivative definition, the key step is to factor out $h$ before taking the limit.
\item The same idea works in general: the binomial theorem for $(x+h)^n$ leads to the power rule $\dfrac{d}{dx}(x^n)=nx^{n-1}$.
\item This is a calculus enrichment problem, but it shows one way algebra and Pascal's triangle support differentiation.
\end{takeaways}
