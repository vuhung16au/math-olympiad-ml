\section{Fundamentals Review}

\subsection{Key Inequalities}
Problem~16 frequently employs fundamental inequalities as building blocks for more complex arguments.

\subsubsection{Arithmetic-Geometric Mean (AM-GM) Inequality}
\begin{theorem}[AM-GM Inequality]
For positive real numbers $a_1, a_2, \ldots, a_n$:
\[\frac{a_1 + a_2 + \cdots + a_n}{n} \geq \sqrt[n]{a_1 a_2 \cdots a_n}\]
Equality occurs if and only if $a_1 = a_2 = \cdots = a_n$.
\end{theorem}

\begin{theorem}[Weighted AM-GM Inequality]
For positive real numbers $a_1, a_2, \ldots, a_n$ and positive weights $w_1, w_2, \ldots, w_n$ with $w_1 + w_2 + \cdots + w_n = 1$:
\[w_1 a_1 + w_2 a_2 + \cdots + w_n a_n \geq a_1^{w_1} a_2^{w_2} \cdots a_n^{w_n}\]
\end{theorem}

\subsubsection{Cauchy-Schwarz Inequality}
\begin{theorem}[Cauchy-Schwarz Inequality]
For vectors $\mathbf{u}$ and $\mathbf{v}$ in $\mathbb{R}^n$:
\[|\mathbf{u} \cdot \mathbf{v}| \leq |\mathbf{u}| |\mathbf{v}|\]
Equivalently, for real numbers $a_1, a_2, \ldots, a_n$ and $b_1, b_2, \ldots, b_n$:
\[\left(\sum_{i=1}^{n} a_i b_i\right)^2 \leq \left(\sum_{i=1}^{n} a_i^2\right)\left(\sum_{i=1}^{n} b_i^2\right)\]
\end{theorem}

\subsection{Complex Number Theory}
\subsubsection{De Moivre's Theorem}
\begin{theorem}[De Moivre's Theorem]
For complex numbers in polar form $z = r(\cos\theta + i\sin\theta)$:
\[z^n = r^n(\cos(n\theta) + i\sin(n\theta))\]
\end{theorem}

\subsubsection{Roots of Unity}
\begin{theorem}[Roots of Unity]
The $n$th roots of unity are given by:
\[\omega_k = e^{2\pi i k/n} = \cos\left(\frac{2\pi k}{n}\right) + i\sin\left(\frac{2\pi k}{n}\right)\]
for $k = 0, 1, 2, \ldots, n-1$.
\end{theorem}

\subsection{Vector Algebra}
\subsubsection{Dot Product}
For vectors $\mathbf{u} = (u_1, u_2, u_3)$ and $\mathbf{v} = (v_1, v_2, v_3)$:
\[\mathbf{u} \cdot \mathbf{v} = u_1v_1 + u_2v_2 + u_3v_3 = |\mathbf{u}||\mathbf{v}|\cos\theta\]

\subsubsection{Cross Product}
\[\mathbf{u} \times \mathbf{v} = \begin{vmatrix} \mathbf{i} & \mathbf{j} & \mathbf{k} \\ u_1 & u_2 & u_3 \\ v_1 & v_2 & v_3 \end{vmatrix}\]
The magnitude $|\mathbf{u} \times \mathbf{v}| = |\mathbf{u}||\mathbf{v}|\sin\theta$ equals the area of the parallelogram formed by $\mathbf{u}$ and $\mathbf{v}$.

\subsection{Polynomial Theory}
\subsubsection{Newton's Identities}
\begin{theorem}[Newton's Identities]
For a polynomial with roots $\alpha_1, \alpha_2, \ldots, \alpha_n$, if $S_k = \sum_{i=1}^n \alpha_i^k$ and $e_k$ are the elementary symmetric polynomials, then:
\[S_k - e_1 S_{k-1} + e_2 S_{k-2} - \cdots + (-1)^{k-1} e_{k-1} S_1 + (-1)^k k e_k = 0\]
\end{theorem}

\subsection{Chebyshev Polynomials (First Kind)}
The Chebyshev polynomials of the first kind $T_n(x)$ are a family of orthogonal polynomials on $[-1,1]$ with the defining trigonometric identity:
\[ T_n(x) = \cos\big(n\arccos x\big). \]

\subsubsection{Recurrence Formula}
They satisfy the three-term recurrence for $n\ge 1$:
\[ T_{n+1}(x) = 2x\,T_n(x) - T_{n-1}(x), \quad T_0(x)=1,\; T_1(x)=x. \]

\subsubsection{Explicit Low-Degree Polynomials}
\[ T_0(x) = 1, \quad T_1(x) = x, \quad T_2(x) = 2x^2 - 1, \quad T_3(x) = 4x^3 - 3x. \]

These identities are frequently useful for transforming certain cubic equations by matching the structure of $T_3(x)$.

\subsection{Optimization Techniques}
\subsubsection{Lagrange Multipliers (Conceptual)}
To optimize $f(x,y,z)$ subject to constraint $g(x,y,z) = 0$, look for points where:
\[\nabla f = \lambda \nabla g\]
This often reduces to applying AM-GM or Cauchy-Schwarz in HSC problems.
Good to know though the technique is not covered at HSC level.
