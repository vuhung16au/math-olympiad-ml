\section*{Problem 6}
From samples/07.tex

(i) Show that the area of triangle $ABC$ with $A, B, C$ on the axes is $S = \frac{1}{2} \sqrt{b^2 c^2 + a^2 c^2 + a^2 b^2}$.

(ii) Let $S_{xy}, S_{yz}, S_{zx}$ be the areas of the projections of $\triangle ABC$ onto the coordinate planes. Prove that $S^2 = S_{xy}^2 + S_{yz}^2 + S_{zx}^2$.

\section*{Problem 7}
From samples/08.tex

Four unit vectors $\mathbf{v}_1, \mathbf{v}_2, \mathbf{v}_3, \mathbf{v}_4$ originate from the origin $O$ and their sum is zero. Let $\theta_{ij}$ be the angle between $\mathbf{v}_i$ and $\mathbf{v}_j$. Show that $\sum_{1 \le i < j \le 4} \cos \theta_{ij} = -2$.

% Problem 8: Complex Systems via Newton Sums
\section*{Problem 8}
From samples/09.tex

Find all complex numbers $z_1, z_2, z_3$ that satisfy the following system of equations simultaneously:
\[
\begin{cases}
    z_1 + z_2 + z_3 = 3 \\
    z_1^2 + z_2^2 + z_3^2 = 3 \\
    z_1^3 + z_2^3 + z_3^3 = 3
\end{cases}
\]

% Problem 9: Unit Circle & Reciprocals
\section*{Problem 9}
From samples/10.tex

Let $z_1, z_2, z_3$ be distinct complex numbers such that:
\[ |z_1| = |z_2| = |z_3| = 1 \]
\[ z_1 + z_2 + z_3 = 1 \]
\[ z_1 z_2 z_3 = 1 \]
Prove that at least one of the complex numbers must be equal to $1$.

% Problem 10: Zero Sum Geometry
\section*{Problem 10}
From samples/11.tex

Three complex numbers $z_1, z_2, z_3$ satisfy:
\[
\begin{cases}
    |z_1| = |z_2| = |z_3| = r & (r > 0) \\
    z_1 + z_2 + z_3 = 0
\end{cases}
\]
Show that $z_1, z_2, z_3$ represent the vertices of an equilateral triangle inscribed in the circle $|z|=r$.

% Problem 11: Irrationality of e
\section*{Problem 11}
% From samples/56.tex

You are given the infinite series definition:
\[ e = \sum_{n=0}^{\infty} \frac{1}{n!} = 1 + \frac{1}{1!} + \frac{1}{2!} + \frac{1}{3!} + \dots \]

\begin{enumerate}
    \item[(i)] Let $S_k = \sum_{n=0}^{k} \frac{1}{n!}$ be the $k$-th partial sum.
    Show that for any integer $k \ge 1$:
    \[ 0 < e - S_k < \frac{1}{k! k} \]
    (Hint: Compare the remainder series to a geometric progression).
    
    \item[(ii)] Suppose, for the sake of contradiction, that $e$ is rational, i.e., $e = \frac{p}{q}$ for positive integers $p, q$.
    Consider the number $X = q! (e - S_q)$.
    Show that $X$ must be an integer.
    
    \item[(iii)] Using the inequality from part (i), show that $0 < X < 1$.
    
    \item[(iv)] Hence, conclude that $e$ is irrational.
\end{enumerate}
