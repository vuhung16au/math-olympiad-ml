% Problem 15: Mandelbrot Escape Criterion
\begin{problem}[Mandelbrot Escape Criterion]
\noindent \textbf{Topic: Complex Dynamics, Inequalities, Mathematical Induction}

\medskip

For a given complex constant $c \in \mathbb{C}$, define the Mandelbrot sequence by
\[
z_0 = 0, \qquad z_{n+1} = z_n^2 + c \quad (n \ge 0).
\]
The Mandelbrot set $\mathcal{M}$ is the set of all $c \in \mathbb{C}$ for which the sequence of moduli $(|z_n|)_{n\ge 0}$ remains bounded.

\begin{enumerate}
    \item[(i)] Suppose $c \in \mathbb{R}$ and $c > \frac{1}{4}$. By induction, prove that for all $n \ge 0$, $z_n \in \mathbb{R}$ and
    \[
    z_{n+1} - z_n = z_n^2 - z_n + c > 0.
    \]
    Hence, deduce that if $c > \frac{1}{4}$ then $c \notin \mathcal{M}$.

    \item[(ii)] Recall the reverse triangle inequality $|u - v| \ge \big| |u| - |v| \big|$. Let $z \in \mathbb{C}$ with $|z| \ge 1$. Prove that
    \[
    |z^2 - z| \ge |z|(|z| - 1).
    \]

    \item[(iii)] Let $c \in \mathbb{C}$ with $|c| > 2$, and define $z_0 = 0, z_{n+1} = z_n^2 + c$. Note that $|z_1| = |c| > 2$. Using the estimate in (ii) as the base idea, prove by induction that for all $n \ge 1$ with $|z_n| \ge 2$ we have
    \[
    |z_{n+1}| \ge |z_n|^2 - |c| \ge |z_n|^2 - |z_n| = |z_n|(|z_n| - 1).
    \]
    Hence, conclude that if $|c| > 2$, then $c \notin \mathcal{M}$.

    \item[(iv)] Determine, using the results above, whether $c = 2 + i$ belongs to $\mathcal{M}$. Justify your answer rigorously.
\end{enumerate}

\begin{hint}
\begin{itemize}
    \item \textbf{(i)} Try completing the square for the quadratic $x^2 - x + c$. What does $c > \frac{1}{4}$ guarantee about its minimum value?
    \item \textbf{(ii)} Factorize $z^2 - z$ into $z(z - 1)$. Apply the multiplicative property of absolute values, then use the reverse triangle inequality on $|z - 1|$.
    \item \textbf{(iii)} Use $|z_{n+1}| = |z_n^2 + c| \ge |z_n|^2 - |c|$ (reverse triangle inequality). When $|z_n| \ge 2$ and $|c| > 2$, show $|z_n| \ge |c|$ to get $|z_n|^2 - |c| \ge |z_n|(|z_n|-1)$. Conclude $|z_{n+1}| \ge |z_n|$ so $|z_n| \ge 2$ for all $n \ge 1$ and the sequence grows.
    \item \textbf{(iv)} Compute $|2 + i|$. Which result from (i)--(iii) applies?
\end{itemize}
\end{hint}

\begin{solution}
\textbf{(i)} Since $z_0=0\in\mathbb{R}$ and $z_{n+1}=z_n^2+c\in\mathbb{R}$ whenever $z_n\in\mathbb{R}$, induction gives $z_n\in\mathbb{R}$ for all $n$. Also
\[
z_{n+1}-z_n=z_n^2-z_n+c=\left(z_n-\frac12\right)^2+c-\frac14.
\]
If $c>\frac14$, then $z_{n+1}-z_n>0$ for every real $z_n$, so $(z_n)$ is strictly increasing. If it were bounded, it would converge to some real limit $L$ with
\[
L=L^2+c \qquad\Rightarrow\qquad L^2-L+c=0.
\]
But for $c>\frac14$, the discriminant is $1-4c<0$, impossible for a real $L$. Hence $(z_n)$ is unbounded, so $c\notin\mathcal M$.

\textbf{(ii)} Since
\[
z^2-z=z(z-1),
\]
we have
\[
|z^2-z|=|z|\,|z-1|\ge |z|\,\bigl||z|-1\bigr|.
\]
If $|z|\ge1$, then $\bigl||z|-1\bigr|=|z|-1$, hence
\[
|z^2-z|\ge |z|(|z|-1).
\]

\textbf{(iii)} Assume $|c|>2$. Then $|z_1|=|c|>2$, and
\[
|z_{n+1}|=|z_n^2+c|\ge |z_n|^2-|c|.
\]
Now suppose $|z_n|\ge |c|$. Then
\[
|z_{n+1}|\ge |z_n|^2-|c|\ge |z_n|^2-|z_n|=|z_n|(|z_n|-1).
\]
If also $|z_n|\ge2$, this gives $|z_{n+1}|\ge |z_n|\ge2$. Since $|z_1|=|c|>2$, induction yields $|z_n|\ge2$ for all $n\ge1$, and in fact the moduli are non-decreasing. Therefore $|z_n|$ grows without bound, so $c\notin\mathcal M$.

\textbf{(iv)} For $c=2+i$,
\[
|c|=|2+i|=\sqrt5>2.
\]
By part (iii), $2+i\notin\mathcal M$.

\begin{takeaways}
\begin{enumerate}
    \item \textbf{Escape Criterion:} If $|c| > 2$, the Mandelbrot sequence escapes to infinity; such $c$ lie outside the Mandelbrot set. The diagram shows $c = 2 + i$ outside the circle of radius 2.
    \item \textbf{Reverse Triangle Inequality:} $|u - v| \ge \big| |u| - |v| \big|$ is essential for modulus estimates in complex dynamics.
    \item \textbf{Completing the Square:} For real $c > \tfrac{1}{4}$, the quadratic $x^2 - x + c$ has no real roots, so no fixed point exists and the sequence diverges.
    \item \textbf{Induction on Modulus:} When $|z_n| \ge 2$, the recurrence $z_{n+1} = z_n^2 + c$ amplifies the modulus; induction shows unbounded growth.
\end{enumerate}
\end{takeaways}
