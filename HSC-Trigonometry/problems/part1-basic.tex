\begin{solution}

The graph of $y = \sin\!\left(x - \dfrac{\pi}{2}\right)$ is the graph of
$y = \sin x$ shifted \textbf{right} by $\dfrac{\pi}{2}$ (a phase shift of
$\dfrac{\pi}{2}$).

\medskip
Using the compound-angle identity:
\[
\sin\!\left(x - \frac{\pi}{2}\right)
  = \sin x\cos\frac{\pi}{2} - \cos x\sin\frac{\pi}{2}
  = \sin x \cdot 0 - \cos x \cdot 1
  = -\cos x.
\]
Hence $y = \sin\!\left(x - \dfrac{\pi}{2}\right) = -\cos x$.

\medskip
\textbf{Key points} on $[-2\pi,\, 2\pi]$:

\medskip
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{c|ccccccccc}
$x$ & $-2\pi$ & $-\dfrac{3\pi}{2}$ & $-\pi$ & $-\dfrac{\pi}{2}$
    & $0$    & $\dfrac{\pi}{2}$   & $\pi$  & $\dfrac{3\pi}{2}$
    & $2\pi$ \\[6pt] \hline
$y$ & $-1$ & $0$ & $1$ & $0$ & $-1$ & $0$ & $1$ & $0$ & $-1$
\end{tabular}
\end{center}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
  \item \textbf{Phase shift:} The graph of $y = \sin(x - c)$ for $c > 0$ is the
        graph of $y = \sin x$ shifted $c$ units to the \emph{right}.

  \item \textbf{Identity connection:}
        $\sin\!\left(x - \dfrac{\pi}{2}\right) = -\cos x$,
        so this curve is also the reflection of $y = \cos x$ in the $x$-axis.

  \item \textbf{Graphical check:} The shifted sine reaches its zeros at
        $x = \dfrac{\pi}{2} + k\pi$ ($k \in \mathbb{Z}$), its maximum of $1$
        at $x = \pi + 2k\pi$, and its minimum of $-1$ at $x = 2k\pi$
        — each exactly $\dfrac{\pi}{2}$ to the right of the corresponding
        feature on $y = \sin x$.
\end{itemize}
\end{takeaways}

\begin{problem}[Sketching a Transformed Cosine]
Consider the function
\[
y = 2\cos\!\left(x - \frac{\pi}{3}\right), \qquad x \in [0, 2\pi].
\]
\begin{enumerate}[(i)]
  \item State the amplitude, period, and phase shift of this function.
  \item Find all $x$-intercepts in $[0, 2\pi]$.
  \item Sketch the graph for $x \in [0, 2\pi]$, clearly marking the intercepts,
        the maximum, and the minimum.
\end{enumerate}
\end{problem}
% ------------------------------------------------------------------

\begin{solution}

\textbf{(i) Key features.}

Comparing $y = 2\cos\!\left(x - \dfrac{\pi}{3}\right)$ with the standard form
$y = A\cos(bx + c) + d$, we read off $A = 2$, $b = 1$, $c = -\pi/3$, $d = 0$:
\[
\text{Amplitude} = |A| = 2, \qquad
\text{Period} = \frac{2\pi}{|b|} = 2\pi, \qquad
\text{Phase shift} = -\frac{c}{b} = \frac{\pi}{3} \text{ (right)}.
\]

\medskip
\textbf{(ii) $x$-intercepts.}

Set $y = 0$:
\[
2\cos\!\left(x - \frac{\pi}{3}\right) = 0
\implies
\cos\!\left(x - \frac{\pi}{3}\right) = 0.
\]
The cosine is zero when its argument equals $\dfrac{\pi}{2} + k\pi$ for integer $k$:
\[
x - \frac{\pi}{3} = \frac{\pi}{2} + k\pi
\implies
x = \frac{\pi}{3} + \frac{\pi}{2} + k\pi = \frac{5\pi}{6} + k\pi.
\]
For $x \in [0, 2\pi]$: $k = 0$ gives $x = \dfrac{5\pi}{6}$, and $k = 1$ gives
$x = \dfrac{5\pi}{6} + \pi = \dfrac{11\pi}{6}$.

The $x$-intercepts are $\boldsymbol{x = \dfrac{5\pi}{6}}$ and $\boldsymbol{x = \dfrac{11\pi}{6}}$.

\medskip
\textbf{Key points on $[0, 2\pi]$:}

\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{c|ccccc}
$x$ & $0$ & $\dfrac{\pi}{3}$ & $\dfrac{5\pi}{6}$ & $\dfrac{4\pi}{3}$ & $\dfrac{11\pi}{6}$ \\[6pt]\hline
$x - \pi/3$ & $-\pi/3$ & $0$ & $\pi/2$ & $\pi$ & $3\pi/2$ \\[4pt]
$y$ & $1$ & $2$ & $0$ & $-2$ & $0$
\end{tabular}
\end{center}

The maximum value of $2$ occurs at $x = \dfrac{\pi}{3}$ (where $\cos 0 = 1$),
and the minimum of $-2$ occurs at $x = \dfrac{4\pi}{3}$ (where $\cos\pi = -1$).
The curve also passes through $(0, 1)$ and $(2\pi, 1)$ as endpoints.

\end{solution}

\begin{takeaways}
\begin{itemize}[leftmargin=*]
  \item For $y = A\cos(bx + c) + d$: amplitude $= |A|$, period $= 2\pi/|b|$,
        phase shift $= -c/b$ (positive means shift right).
  \item To find $x$-intercepts, set the cosine expression equal to zero and use the
        general solution $\cos u = 0 \Rightarrow u = \pi/2 + k\pi$.
  \item A phase shift of $\pi/3$ moves every feature of $y = 2\cos x$ exactly
        $\pi/3$ units to the right; the shape and scale are unchanged.
\end{itemize}
\end{takeaways}
