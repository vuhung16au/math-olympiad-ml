\section*{The Reciprocal Sum Inequality Problem}

\subsection*{1. Problem Statement}

\textbf{i. Standard AM-HM Inequality:}
Prove that for three positive real numbers $a, b, c$, the Arithmetic Mean (AM) is greater than or equal to the Harmonic Mean (HM):
$$\frac{a+b+c}{3} \ge \frac{3}{\frac{1}{a} + \frac{1}{b} + \frac{1}{c}}$$

\textbf{ii. Reciprocal Sum Inequality:}
Hence, show that for positive real numbers $a, b, c$:
$$(a+b+c)\left(\frac{1}{a+b} + \frac{1}{b+c} + \frac{1}{c+a}\right) \ge \frac{9}{2}$$

\subsection*{2. Hints}

\begin{enumerate}
    \item \textbf{Hint for i (AM-HM):} Use the Arithmetic Mean - Geometric Mean (AM-GM) inequality, $\frac{x+y+z}{3} \ge \sqrt[3]{xyz}$.
    \item \textbf{Hint for i (Proof Technique):} Apply AM-GM to the set $\{a, b, c\}$ and the set of reciprocals $\left\{\frac{1}{a}, \frac{1}{b}, \frac{1}{c}\right\}$. Then, multiply the two resulting inequalities.
    \item \textbf{Hint for ii (Substitution):} After proving that $\left(x+y+z\right) \left(\frac{1}{x} + \frac{1}{y} + \frac{1}{z}\right) \ge 9$, apply a strategic substitution to relate this form to the expression in part (ii). Let the variables be $x' = a+b$, $y' = b+c$, and $z' = c+a$.
\end{enumerate}

\subsection*{3. Solution (Outlined)}

\textbf{i. Proof of the AM-HM Inequality:}
\begin{itemize}
    \item \textbf{Step 1: AM-GM on $a, b, c$ and $\frac{1}{a}, \frac{1}{b}, \frac{1}{c}$}
    $$a+b+c \ge 3\sqrt[3]{abc} \quad \text{and} \quad \frac{1}{a} + \frac{1}{b} + \frac{1}{c} \ge 3\sqrt[3]{\frac{1}{abc}}$$
    \item \textbf{Step 2: Multiplication}
    Multiply the two inequalities from Step 1:
    $$(a+b+c)\left(\frac{1}{a} + \frac{1}{b} + \frac{1}{c}\right) \ge 9 \left(\sqrt[3]{abc} \cdot \sqrt[3]{\frac{1}{abc}}\right)$$
    $$(a+b+c)\left(\frac{1}{a} + \frac{1}{b} + \frac{1}{c}\right) \ge 9 \cdot 1$$
    \item \textbf{Step 3: Rearrangement}
    Divide by 9 and $\left(\frac{1}{a} + \frac{1}{b} + \frac{1}{c}\right)$ to obtain the standard AM-HM form:
    $$\frac{a+b+c}{3} \ge \frac{3}{\frac{1}{a} + \frac{1}{b} + \frac{1}{c}}$$
\end{itemize}

\textbf{ii. Showing the Reciprocal Sum Inequality:}
\begin{itemize}
    \item \textbf{Step 1: Apply the AM-HM Result in the Form $\ge 9$}
    Use the result: $\left(x+y+z\right) \left(\frac{1}{x} + \frac{1}{y} + \frac{1}{z}\right) \ge 9$.
    \item \textbf{Step 2: Substitution}
    Let $x = a+b$, $y = b+c$, and $z = c+a$. Substitute these into the inequality:
    $$((a+b) + (b+c) + (c+a)) \left( \frac{1}{a+b} + \frac{1}{b+c} + \frac{1}{c+a} \right) \ge 9$$
    \item \textbf{Step 3: Simplification}
    Simplify the first factor:
    $$(2a + 2b + 2c) \left( \frac{1}{a+b} + \frac{1}{b+c} + \frac{1}{c+a} \right) \ge 9$$
    $$2(a+b+c) \left( \frac{1}{a+b} + \frac{1}{b+c} + \frac{1}{c+a} \right) \ge 9$$
    \item \textbf{Step 4: Final Rearrangement}
    Divide by 2 to get the final result:
    $$(a+b+c) \left( \frac{1}{a+b} + \frac{1}{b+c} + \frac{1}{c+a} \right) \ge \frac{9}{2}$$
\end{itemize}

\subsection*{4. Takeaways}

\begin{itemize}
    \item \textbf{Powerful Identity:} The inequality $\left(x+y+z\right) \left(\frac{1}{x} + \frac{1}{y} + \frac{1}{z}\right) \ge 9$ is a critical application of AM-GM/AM-HM, useful for solving many competition-style problems.
    \item \textbf{Strategic Substitution:} Part (ii) demonstrates how a complex inequality can be solved by recognizing a core inequality structure (the AM-HM form) and making a clever substitution (e.g., $x=a+b$).
    \item \textbf{Equality Condition:} The equality holds if and only if $a=b=c$. This makes the two factors in the LHS of part (ii) become $3a$ and $\frac{3}{2a}$, giving $3a \cdot \frac{3}{2a} = \frac{9}{2}$.
\end{itemize}
