\subsection{Inequality Fundamentals}

\subsubsection{Basic Properties}
For real numbers $a$, $b$, $c$, and $d$, the following properties hold:
\begin{itemize}[leftmargin=*]
    \item \textbf{Transitivity:} $a > b$ and $b > c \Rightarrow a > c$
    \item \textbf{Multiplication:} $a > b$ and $c > 0 \Rightarrow ac > bc$; but $c < 0 \Rightarrow ac < bc$
    \item \textbf{Product rule:} $a > b > 0$ and $c > d > 0 \Rightarrow ac > bd$
    \item \textbf{Reciprocals:} $a > b > 0 \Rightarrow \dfrac{1}{a} < \dfrac{1}{b}$
    \item \textbf{Non-negativity:} For any real $x$, we have $x^2 \ge 0$
    \item \textbf{Absolute values:} $|a| \ge a$ and $|x| + |y| \ge |x + y|$ (triangle inequality)
    \item \textbf{Sum of squares:} $a^2 + b^2 \ge 0$ with equality if and only if $a = b = 0$
\end{itemize}

\subsubsection{Key Theorems}

\paragraph{AM-GM Inequality (Two Variables).}
For non-negative real numbers $x$ and $y$:
\[
\frac{x + y}{2} \ge \sqrt{xy}
\]
with equality if and only if $x = y$.

\paragraph{AM-GM Inequality (Three Variables).}
For non-negative real numbers $x$, $y$, and $z$:
\[
\frac{x + y + z}{3} \ge \sqrt[3]{xyz}
\]
with equality if and only if $x = y = z$.

\paragraph{Cauchy-Schwarz Inequality.}
For real numbers $a_1, a_2, \ldots, a_n$ and $b_1, b_2, \ldots, b_n$:
\[
\left(\sum_{i=1}^{n} a_i b_i\right)^2 \le \left(\sum_{i=1}^{n} a_i^2\right)\left(\sum_{i=1}^{n} b_i^2\right)
\]
with equality if and only if the sequences are proportional.

\paragraph{Triangle Inequality (Real Numbers).}
For real numbers $a$ and $b$:
\[
|a + b| \le |a| + |b|
\]

\paragraph{Triangle Inequality (Complex Numbers).}
For complex numbers $z$ and $w$:
\[
|z + w| \le |z| + |w|
\]

\subsection{Worked Examples}

\paragraph{Example 1: Basic AM-GM Application.}
Prove that for positive real numbers $a$ and $b$, we have $a + b \ge 2\sqrt{ab}$.

\textit{Solution:}
By the AM-GM inequality with $x = a$ and $y = b$:
\[
\frac{a + b}{2} \ge \sqrt{ab}
\]
Multiplying both sides by 2 yields $a + b \ge 2\sqrt{ab}$, with equality when $a = b$. \qed

\paragraph{Example 2: Cauchy-Schwarz with Constraint.}
Given $x^2 + y^2 = 1$, find the maximum value of $3x + 4y$.

\textit{Solution:}
By Cauchy-Schwarz inequality:
\[
(3x + 4y)^2 \le (3^2 + 4^2)(x^2 + y^2) = 25 \cdot 1 = 25
\]
Therefore $3x + 4y \le 5$, with equality when $(x, y) = \left(\frac{3}{5}, \frac{4}{5}\right)$. The maximum value is $5$. \qed

\paragraph{Example 3: Triangle Inequality.}
Prove that for any complex number $z$ with $|z| = 1$, we have $|z^2 + z + 1| \le 3$.

\textit{Solution:}
Using the triangle inequality repeatedly:
\[
|z^2 + z + 1| \le |z^2| + |z| + |1| = 1 + 1 + 1 = 3
\]
with equality when $z = 1$. \qed

\paragraph{Example 4: Integration Inequality.}
Prove that $\displaystyle\int_0^1 x^2 \, dx < \int_0^1 x \, dx$.

\textit{Solution:}
For $x \in [0, 1]$, we have $x^2 \le x$ (with equality only at $x = 0$ and $x = 1$). Therefore:
\[
\int_0^1 x^2 \, dx < \int_0^1 x \, dx
\]
Computing: $\dfrac{1}{3} < \dfrac{1}{2}$. \qed

\paragraph{Example 5: Induction with Inequality.}
Prove by induction that $2^n > n$ for all integers $n \ge 1$.

\textit{Solution:}
\textit{Base case ($n = 1$):} $2^1 = 2 > 1$. \checkmark

\textit{Inductive step:} Assume $2^k > k$ for some $k \ge 1$. Then:
\[
2^{k+1} = 2 \cdot 2^k > 2k
\]
Since $k \ge 1$, we have $2k = k + k \ge k + 1$. Therefore $2^{k+1} > k + 1$. By induction, the result holds for all $n \ge 1$. \qed
