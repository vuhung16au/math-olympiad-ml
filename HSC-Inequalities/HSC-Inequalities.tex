\subsection{Inequality Fundamentals}

\subsubsection{Basic Properties}
For real numbers $a$, $b$, $c$, and $d$, the following properties hold:
\begin{itemize}[leftmargin=*]
    \item \textbf{Transitivity:} $a > b$ and $b > c \Rightarrow a > c$
    \item \textbf{Multiplication:} $a > b$ and $c > 0 \Rightarrow ac > bc$; but $c < 0 \Rightarrow ac < bc$
    \item \textbf{Product rule:} $a > b > 0$ and $c > d > 0 \Rightarrow ac > bd$
    \item \textbf{Reciprocals:} $a > b > 0 \Rightarrow \dfrac{1}{a} < \dfrac{1}{b}$
    \item \textbf{Non-negativity:} For any real $x$, we have $x^2 \ge 0$
    \item \textbf{Absolute values:} $|a| \ge a$ and $|x| + |y| \ge |x + y|$ (triangle inequality)
    \item \textbf{Sum of squares:} $a^2 + b^2 \ge 0$ with equality if and only if $a = b = 0$
\end{itemize}

\subsubsection{Key Theorems}

\begin{theorembox}[title={\bfseries AM-GM Inequality (Two Variables)}]
For non-negative real numbers $x$ and $y$:
\[
\frac{x + y}{2} \ge \sqrt{xy}
\]
with equality if and only if $x = y$.
\end{theorembox}

\begin{theorembox}[title={\bfseries AM-GM Inequality (Three Variables)}]
For non-negative real numbers $x$, $y$, and $z$:
\[
\frac{x + y + z}{3} \ge \sqrt[3]{xyz}
\]
with equality if and only if $x = y = z$.
\end{theorembox}

\begin{remark}[Two Approaches to Proving General Inequality]AM-GM]
The AM-GM inequality states that for positive reals $x_1, x_2, \ldots, x_n$:
\[ \frac{x_1 + x_2 + \cdots + x_n}{n} \geq \sqrt[n]{x_1 x_2 \cdots x_n} \]

\textbf{(a) Proof by Induction on $n$:}
\begin{enumerate}
    \item \textit{Base case:} $n=2$ follows from $(\sqrt{x_1} - \sqrt{x_2})^2 \geq 0$
    \item \textit{Forward-backward step:} Prove $n=2^k \Rightarrow n=2^{k+1}$ by grouping pairs
    \item \textit{Backward step:} Show $n=k \Rightarrow n=k-1$ by setting $x_k = \frac{x_1 + \cdots + x_{k-1}}{k-1}$ and applying the $n=k$ case
\end{enumerate}

\begin{theorembox}[title={\bfseries Cauchy-Schwarz Inequality}]
For real numbers $a_1, a_2, \ldots, a_n$ and $b_1, b_2, \ldots, b_n$:
\[
\left(\sum_{i=1}^{n} a_i b_i\right)^2 \le \left(\sum_{i=1}^{n} a_i^2\right)\left(\sum_{i=1}^{n} b_i^2\right)
\]
with equality if and only if the sequences are proportional.
\end{theorembox}

\textit{Proof outline (discriminant method):}
Let $A=\sum a_i^2$, $B=\sum b_i^2$, $C=\sum a_i b_i$, and consider
$$P(t)=\sum_{i=1}^{n}(a_i^2 t^2-2a_i b_i t+b_i^2)=At^2-2Ct+B\ge 0\ \text{for all }t.$$ Since $P(t)$ is non-negative, its discriminant satisfies $\Delta=(-2C)^2-4AB\le 0$, hence $C^2\le AB$. Substituting back yields $\left(\sum a_i b_i\right)^2\le\left(\sum a_i^2\right)\left(\sum b_i^2\right)$. Equality holds iff $\Delta=0$, i.e., there exists $t_0$ with $a_i t_0=b_i$ for all $i$ (proportional sequences).

\begin{definitionbox}[title={\bfseries Convex Function}]
A function $f$ is \textbf{convex} on an interval $I$ if $f''(x) \ge 0$ for all $x \in I$ (its graph is concave up). If $f''(x) \le 0$, then $f$ is \textbf{concave}.
\end{definitionbox}

\begin{theorembox}[title={\bfseries Jensen's Inequality}]
Let $f$ be a convex function on an interval $I$. 
For points $x_1,\dots,x_n\in I$ and non-negative weights 
$\lambda_1,\dots,\lambda_n$ with $\sum_{i=1}^n\lambda_i=1$, Jensen's inequality states
\[
f\Big(\sum_{i=1}^n\lambda_i x_i\Big) \le \sum_{i=1}^n\lambda_i f(x_i).
\]
In the equal-weight case ($\lambda_i = \tfrac{1}{n}$), this specializes to
\[
 f\Big(\tfrac{x_1 + x_2 + \cdots + x_n}{n}\Big) \le \tfrac{f(x_1) + f(x_2) + \cdots + f(x_n)}{n},
\]
"the function of the average is less than or equal to the average of the function."
If $f$ is concave, the inequality is reversed. 
Equality holds when the $x_i$ are all equal or when $f$ is linear on the convex hull of the points.
\end{theorembox}


\begin{remark}
\noindent\begin{minipage}[t]{0.49\linewidth}\vspace{0pt}
Jensen's inequality generalizes the statement that the secant line of a convex function lies \textit{above} the graph of the function, which is Jensen's inequality for two points: the secant line consists of weighted means of the convex function (for $t \in [0,1]$),
\[
 t f(x_1) + (1 - t) f(x_2),
\]
while the graph of the function is the convex function of the weighted means,
\[
 f(t x_1 + (1 - t) x_2).
\]
Thus, Jensen's inequality in this case is
\[
 f(t x_1 + (1 - t) x_2) \leq t f(x_1) + (1 - t) f(x_2).
\]

\begin{theorembox}[title={\bfseries Triangle Inequality (Real Numbers)}]
For real numbers $a$ and $b$:
\[
|a + b| \le |a| + |b|
\]
\end{theorembox}

\begin{theorembox}[title={\bfseries Triangle Inequality (Complex Numbers)}]
For complex numbers $z$ and $w$:
\[
|z + w| \le |z| + |w|
\]
\end{theorembox}

Notes: (i) The polygonal inequality generalizes the triangle inequality to sums of more than two terms by repeated application. (ii) The geometric interpretation is that in any triangle, the length of one side is less than or equal to the sum of the lengths of the other two sides, which can be applied in 2-D and 3-D geometry problems.

\subsection{Worked Examples}

\begin{examplebox}[title={\bfseries Example~\thetcbcounter: Basic AM-GM Application}]
Prove that for positive real numbers $a$ and $b$, we have $a + b \ge 2\sqrt{ab}$.

\end{examplebox}

\begin{solution}
By the AM-GM inequality with $x = a$ and $y = b$:
\[
\frac{a + b}{2} \ge \sqrt{ab}
\]
Multiplying both sides by 2 yields $a + b \ge 2\sqrt{ab}$, with equality when $a = b$. \qed
\end{solution}

\begin{examplebox}[title={\bfseries Example~\thetcbcounter: Bernoulli's Inequality via Jensen}]
Prove Bernoulli's inequality: for $x\ge -1$ and $r\ge 1$,
\[
  (1+x)^r \ge 1 + r x.
\]

\end{examplebox}

\begin{solution}
Consider $f(t) = (1+t)^r$ on $[-1,\infty)$. For $r\ge 1$, we have
\[
  f''(t) = r(r-1)(1+t)^{r-2} \ge 0,
\]
so $f$ is convex. A standard corollary of Jensen's inequality (supporting line/tangent property for convex functions) states that for convex, differentiable $f$ and any $y$,\footnote{This follows from convexity: the graph of a convex function lies above every tangent line; equivalently, it is a consequence of Jensen when applied to a degenerate two-point distribution and letting one weight tend to zero.}
\[
  f(y) \ge f(x) + f'(x)(y-x) \quad \text{for all }x,y.
\]
Taking $x=0$ and $y=x$ (note the re-use of $x$ as the variable),
\[
  (1+x)^r = f(x) \ge f(0) + f'(0)\,x = 1 + r x,
\]
which is Bernoulli's inequality. Equality occurs at $x=0$. \qed
\end{solution}

\begin{takeaways}
  \item Use convexity of $f(t)=(1+t)^r$ for $r\ge 1$ and the tangent-line corollary of Jensen at $t=0$ to obtain $(1+x)^r\ge 1+rx$; equality at $x=0$.
\end{takeaways}

\begin{examplebox}[title={\bfseries Example~\thetcbcounter: Cauchy-Schwarz with Constraint}]
Given $x^2 + y^2 = 1$, find the maximum value of $3x + 4y$.

\end{examplebox}

\begin{solution}
By Cauchy-Schwarz inequality:
\[
(3x + 4y)^2 \le (3^2 + 4^2)(x^2 + y^2) = 25 \cdot 1 = 25
\]
Therefore $3x + 4y \le 5$, with equality when $(x, y) = \left(\frac{3}{5}, \frac{4}{5}\right)$. The maximum value is $5$. \qed
\end{solution}

\begin{examplebox}[title={\bfseries Example~\thetcbcounter: Triangle Inequality}]
Prove that for any complex number $z$ with $|z| = 1$, we have $|z^2 + z + 1| \le 3$.

\end{examplebox}

\begin{solution}
Using the triangle inequality repeatedly:
\[
|z^2 + z + 1| \le |z^2| + |z| + |1| = 1 + 1 + 1 = 3
\]
with equality when $z = 1$. \qed
\end{solution}

\begin{examplebox}[title={\bfseries Example~\thetcbcounter: Integration Inequality}]
Prove that $\displaystyle\int_0^1 x^2 \, dx < \int_0^1 x \, dx$.

\end{examplebox}

\begin{solution}
For $x \in [0, 1]$, we have $x^2 \le x$ (with equality only at $x = 0$ and $x = 1$). Therefore:
\[
\int_0^1 x^2 \, dx < \int_0^1 x \, dx
\]
\end{solution}


\begin{examplebox}[title={\bfseries Example~\thetcbcounter: Induction with Inequality}]
Prove by induction that $2^n > n$ for all integers $n \ge 1$.

\end{examplebox}

\begin{solution}
  extit{Base case ($n = 1$):} $2^1 = 2 > 1$. \checkmark

  extit{Inductive step:} Assume $2^k > k$ for some $k \ge 1$. Then:
\[
2^{k+1} = 2 \cdot 2^k > 2k
\]
Since $k \ge 1$, we have $2k = k + k \ge k + 1$. Therefore $2^{k+1} > k + 1$. By induction, the result holds for all $n \ge 1$. \qed
\end{solution}

\begin{examplebox}[title={\bfseries Example~\thetcbcounter: AM-GM via Jensen}]
Prove that for positive real numbers $x_1, x_2, \ldots, x_n$,
\[
  \frac{x_1 + x_2 + \cdots + x_n}{n} \ge \sqrt[n]{x_1 x_2 \cdots x_n}.
\]

\end{examplebox}

\begin{solution}
Consider the function $f(x) = -\ln(x)$, which is convex for $x > 0$ because $f''(x) = \tfrac{1}{x^2} > 0$.
By Jensen's inequality with equal weights $\lambda_i = \tfrac{1}{n}$,
\[
  f\Big(\tfrac{x_1 + x_2 + \cdots + x_n}{n}\Big) \le \tfrac{f(x_1) + f(x_2) + \cdots + f(x_n)}{n}.
\]
Substituting $f(x) = -\ln(x)$ gives
\[
  -\ln\Big(\tfrac{x_1 + \cdots + x_n}{n}\Big) \le -\tfrac{1}{n}\sum_{i=1}^n \ln(x_i) = -\ln\Big((x_1 x_2 \cdots x_n)^{1/n}\Big).
\]
Multiplying both sides by $-1$ reverses the inequality and exponentiating yields the AM-GM inequality. \qed
\end{solution}

\begin{takeaways}
  \item Choose convex $f(x)=-\ln x$ and equal weights $\lambda_i=\tfrac{1}{n}$; equality occurs when all $x_i$ are equal.
\end{takeaways}

\begin{examplebox}[title={\bfseries Example~\thetcbcounter: Cauchy--Schwarz via Jensen}]
Use Jensen's inequality to prove the Cauchy--Schwarz inequality: for real numbers $a_1,\dots,a_n$ and $b_1,\dots,b_n$,
\[
  \Big(\sum_{i=1}^n a_i b_i\Big)^2 \le \Big(\sum_{i=1}^n a_i^2\Big)\Big(\sum_{i=1}^n b_i^2\Big).
\]

\end{examplebox}

\begin{solution}
Assume not all $b_i$ are zero. Consider the convex function $f(t)=t^2$ on $\mathbb{R}$. Define weights and points
\[
  \lambda_i = \frac{b_i^2}{\sum_{j=1}^n b_j^2}\quad(\lambda_i\ge0,\ \sum \lambda_i=1),\qquad x_i = \frac{a_i}{b_i}\ (\text{for } b_i\ne0).
\]
If some $b_i=0$, then $a_i b_i=0$ and the terms do not affect either side of Cauchy--Schwarz, so we may restrict to indices with $b_i\ne0$.
By Jensen's inequality for the convex function $f$,
\[
  f\!\left(\sum_{i=1}^n \lambda_i x_i\right) \le \sum_{i=1}^n \lambda_i f(x_i).
\]
Substituting $f(t)=t^2$, $\lambda_i$ and $x_i$ gives
\[
  \left(\sum_{i=1}^n \frac{b_i^2}{\sum_{j} b_j^2}\cdot \frac{a_i}{b_i}\right)^2 \le \sum_{i=1}^n \frac{b_i^2}{\sum_{j} b_j^2}\cdot \left(\frac{a_i}{b_i}\right)^2.
\]
Simplifying,
\[
  \left(\frac{\sum_{i=1}^n a_i b_i}{\sum_{j=1}^n b_j^2}\right)^2 \le \frac{\sum_{i=1}^n a_i^2}{\sum_{j=1}^n b_j^2}.
\]
Multiplying both sides by $(\sum_{j=1}^n b_j^2)^2$ yields
\[
  \Big(\sum_{i=1}^n a_i b_i\Big)^2 \le \Big(\sum_{i=1}^n a_i^2\Big)\Big(\sum_{i=1}^n b_i^2\Big),
\]
which is the Cauchy--Schwarz inequality. Equality holds if and only if $x_i=\frac{a_i}{b_i}$ are all equal, i.e., the sequences are proportional. \qed
\end{solution}

\begin{takeaways}
  \item Apply Jensen to $f(t)=t^2$ with points $x_i=a_i/b_i$ and weights $\lambda_i=\dfrac{b_i^2}{\sum_j b_j^2}$; equality occurs when $a_i/b_i$ is constant (sequences proportional).
\end{takeaways}
