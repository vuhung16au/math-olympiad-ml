\subsection{Solution to Problem 28: Induction Proof for Series}

\begin{solution}
We prove by induction that
\[
\frac{1}{2^2} + \frac{1}{3^2} + \cdots + \frac{1}{n^2} < \frac{n-1}{n}
\quad \text{for all integers } n \ge 2.
\]

\textbf{Base case ($n = 2$):} The left-hand side is $\dfrac{1}{4} = 0.25$ and the right-hand side is $\dfrac{1}{2} = 0.5$, so $0.25 < 0.5$ \checkmark.

\textbf{Inductive step:} Assume the statement holds for $n = k \ge 2$:
\[
\frac{1}{2^2} + \frac{1}{3^2} + \cdots + \frac{1}{k^2} < \frac{k-1}{k}.
\]
For $n = k+1$ we need
\[
\frac{1}{2^2} + \frac{1}{3^2} + \cdots + \frac{1}{k^2} + \frac{1}{(k+1)^2} < \frac{k}{k+1}.
\]
By the inductive hypothesis, it suffices to show
\[
\frac{k-1}{k} + \frac{1}{(k+1)^2} < \frac{k}{k+1}.
\]
Rearranging,
\[
\frac{k-1}{k} < \frac{k}{k+1} - \frac{1}{(k+1)^2}
= \frac{k(k+1) - 1}{(k+1)^2}
= \frac{k^2 + k - 1}{(k+1)^2}.
\]
Cross-multiplying (all denominators are positive for $k \ge 2$):
\[
(k-1)(k+1)^2 < k(k^2 + k - 1).
\]
Expanding the left-hand side,
\[
(k-1)(k^2 + 2k + 1) = k^3 + k^2 - k - 1.
\]
So we need $k^3 + k^2 - k - 1 < k^3 + k^2 - k$, which simplifies to $-1 < 0$ \checkmark.

Therefore, by mathematical induction, the inequality holds for all $n \ge 2$.
\end{solution}

\begin{takeaways}
\begin{itemize}
    \item Base case: Verify the smallest value $n = 2$
    \item Inductive step: Assume the result for $k$, then add the next term for $k+1$
    \item Algebraic manipulation: Reduce the inductive step to a simple inequality
\end{itemize}
\end{takeaways}
