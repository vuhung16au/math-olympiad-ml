\subsection{Solution to Problem 26: Complex Numbers and Equilateral Triangle}

\begin{solution}
	extbf{(i)} Since $z_2 = \left(\cos\frac{\pi}{3} + i\sin\frac{\pi}{3}\right)z_1$, multiplying by $\cos\frac{\pi}{3} + i\sin\frac{\pi}{3}$ rotates $z_1$ by $60^\circ$ about the origin.

So $|z_2| = |z_1|$ (rotation preserves modulus) and $\text{Arg}(z_2) = \text{Arg}(z_1) + \pi/3$.

Since $O$ is at the origin, $|OA| = |z_1|$, $|OB| = |z_2| = |z_1|$, and the angle $AOB = \pi/3$.

Therefore triangle $OAB$ has two equal sides ($OA = OB$) and the included angle is $60^\circ$, making it equilateral.

	extbf{(ii)} We have $z_2 = \left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right)z_1$.

Then: $z_1^2 + z_2^2 = z_1^2 + \left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right)^2 z_1^2 = z_1^2\!\left[1 + \left(-\frac{1}{2} + \frac{i\sqrt{3}}{2}\right)\right]$ because $\left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right)^2 = \cos\frac{2\pi}{3} + i\sin\frac{2\pi}{3} = -\frac{1}{2} + \frac{i\sqrt{3}}{2}$.

Simplifying the bracket gives $\frac{1}{2} + \frac{i\sqrt{3}}{2}$, so $z_1^2 + z_2^2 = \left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right) z_1^2$.

And: $z_1 z_2 = z_1 \cdot \left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right) z_1 = \left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right) z_1^2$.

Hence $z_1^2 + z_2^2 = z_1 z_2$.
\end{solution}

\begin{takeaways}
\begin{itemize}
    \item Rotation: Multiplying by $\cos\frac{\pi}{3} + i\sin\frac{\pi}{3}$ rotates by $60^\circ$
    \item Equilateral triangle: Two equal sides with $60^\circ$ angle implies equilateral
    \item Complex algebra: Use $\left(\cos\frac{\pi}{3} + i\sin\frac{\pi}{3}\right)^2 = -\frac{1}{2} + \frac{i\sqrt{3}}{2}$
\end{itemize}
\end{takeaways}
