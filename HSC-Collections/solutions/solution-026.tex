\subsection{Solution to Problem 26: Complex Numbers and Equilateral Triangle}

\begin{solution}
\textbf{(i)} We are given
\[
z_2 = \left(\cos\frac{\pi}{3} + i\sin\frac{\pi}{3}\right)z_1.
\]
Multiplying by $\cos\dfrac{\pi}{3} + i\sin\dfrac{\pi}{3}$ rotates $z_1$ through $\dfrac{\pi}{3}$ about the origin, so
\[
|z_2| = |z_1| \qquad \text{and} \qquad \text{Arg}(z_2) = \text{Arg}(z_1) + \frac{\pi}{3}.
\]
Since $O$ is the origin, $|OA| = |z_1|$, $|OB| = |z_2| = |z_1|$, and $\angle AOB = \dfrac{\pi}{3}$.
Triangle $OAB$ therefore has two equal sides and included angle $60^\circ$, so it is equilateral.

\textbf{(ii)} Using $z_2 = \left(\dfrac{1}{2} + \dfrac{i\sqrt{3}}{2}\right)z_1$, first compute
\[
\left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right)^2
= \cos\frac{2\pi}{3} + i\sin\frac{2\pi}{3}
= -\frac{1}{2} + \frac{i\sqrt{3}}{2}.
\]
Then
\begin{align*}
z_1^2 + z_2^2
&= z_1^2 + \left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right)^2 z_1^2 \\
&= z_1^2\left[1 + \left(-\frac{1}{2} + \frac{i\sqrt{3}}{2}\right)\right] \\
&= z_1^2\left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right).
\end{align*}
Also,
\[
z_1 z_2 = z_1 \cdot \left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right) z_1
= \left(\frac{1}{2} + \frac{i\sqrt{3}}{2}\right) z_1^2.
\]
Hence $z_1^2 + z_2^2 = z_1 z_2$.
\end{solution}

\begin{takeaways}
\begin{itemize}
    \item Rotation: Multiplying by $\cos\dfrac{\pi}{3} + i\sin\dfrac{\pi}{3}$ rotates through $60^\circ$
    \item Equilateral triangle: Two equal sides with a $60^\circ$ included angle imply equilateral
    \item Complex algebra: Square the rotation factor using De Moivre's theorem
\end{itemize}
\end{takeaways}
