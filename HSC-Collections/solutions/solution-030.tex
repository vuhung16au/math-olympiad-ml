\subsection{Solution to Problem 30: Proposition Logic}

\begin{solution}
\textbf{(i)} We prove the implication directly. Suppose $k+1$ is divisible by $3$. Then we can write
\[
k+1 = 3m \qquad \text{for some integer } m,
\]
so $k = 3m - 1$. Substituting into $k^3 + 1$,
\begin{align*}
k^3 + 1 &= (3m - 1)^3 + 1 \\
&= 27m^3 - 27m^2 + 9m - 1 + 1 \\
&= 27m^3 - 27m^2 + 9m \\
&= 9m(3m^2 - 3m + 1).
\end{align*}
The factor $9m$ is divisible by $3$, so $k^3 + 1$ is divisible by $3$. Hence proposition $P$ is true.

\textbf{(ii)} The contrapositive of `If $A$, then $B$' is `If not $B$, then not $A$'. So the contrapositive of $P$ is:
\[
\text{``If $k^3+1$ is not divisible by $3$, then $k+1$ is not divisible by $3$.''}
\]
Since a statement and its contrapositive are logically equivalent, this is also true.

\textbf{(iii)} The converse of `If $A$, then $B$' is `If $B$, then $A$'. So the converse of $P$ is:
\[
\text{``If $k^3+1$ is divisible by $3$, then $k+1$ is divisible by $3$.''}
\]
To decide whether it is true, work modulo $3$. If $k^3 + 1$ is divisible by $3$, then
\[
k^3 \equiv -1 \equiv 2 \pmod{3}.
\]
The possible residues of cubes modulo $3$ are
\[
0^3 \equiv 0,\qquad 1^3 \equiv 1,\qquad 2^3 \equiv 8 \equiv 2 \pmod{3}.
\]
So $k^3 \equiv 2 \pmod{3}$ forces $k \equiv 2 \pmod{3}$, and therefore
\[
k + 1 \equiv 2 + 1 \equiv 0 \pmod{3}.
\]
Thus $k+1$ is divisible by $3$, and the converse is \textbf{true}.
\end{solution}

\begin{takeaways}
\begin{itemize}
    \item Direct proof: Substitute $k = 3m - 1$ and expand to show divisibility by $3$
    \item Contrapositive: Negate both the hypothesis and the conclusion
    \item Converse: Swap hypothesis and conclusion; modular arithmetic checks divisibility
\end{itemize}
\end{takeaways}
