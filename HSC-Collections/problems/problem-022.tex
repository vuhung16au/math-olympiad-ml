\subsection{Problem 22: Mechanics with Ropes and Forces}

\begin{problem}
A machine $P$ of mass $M$ kg is being lifted using two ropes. Rope $T_1$ is attached to the ceiling $C$ and rope $T_2$ is attached to the floor $F$. The ropes make angles $\theta$ and $\phi$ with the horizontal respectively, as shown in the diagram below.

\vspace{0.5em}
\noindent
\begin{center}
\begin{tikzpicture}[scale=1.5, >=stealth]
    % Ceiling and Floor
    \draw[thick] (-3.5, 4) -- (6, 4);
    \draw[thick] (-3.5, 0) -- (6, 0);
    
    % Total height height dimension 'h'
    \draw[<->, >=stealth, thin] (-2.8, 0) -- (-2.8, 4) node[midway, fill=white] {$h$};
    
    % Point coordinates
    \coordinate (C) at (-1.8, 4);
    \coordinate (P) at (0, 1.6);
    \coordinate (F) at (3.6, 0);
    
    % Shaded load triangle at P
    \draw[fill=gray!30, draw=black] (0, 1.6) -- (-0.4, 0.9) -- (0.4, 0.9) -- cycle;
    
    % Downward weight force Mg
    \draw[->, >=stealth, thick] (0, 0.9) -- (0, -0.4) node[below] {$Mg$};
    
    % Ropes T1 and T2 with tension arrows
    \draw[->, >=stealth, thick] (P) -- (C);
    \draw[->, >=stealth, thick] (P) -- (F);
    
    % Point dots
    \fill (C) circle (2pt) node[above left] {$C$};
    \fill (F) circle (2pt) node[below right] {$F$};
    \fill (P) circle (2pt) node[left, xshift=-2pt] {$P$};
    
    % Dashed vertical line passing through P
    \draw[dashed, gray!80] (0, -0.5) -- (0, 4.5);
    
    % Dimension line 'd' (horizontal top)
    \draw[thin, gray!60] (-1.8, 4) -- (-1.8, 4.5);
    \draw[<->, >=stealth, thin] (-1.8, 4.3) -- (0, 4.3) node[midway, fill=white] {$d$};
    
    % Dimension line '2d' (horizontal bottom)
    \draw[thin, gray!60] (3.6, 0) -- (3.6, -0.5);
    \draw[<->, >=stealth, thin] (0, -0.4) -- (3.6, -0.4) node[midway, fill=white] {$2d$};
    
    % Dimension line 'l' (vertical)
    \draw[thin, gray!60] (0, 1.6) -- (2.0, 1.6);
    \draw[<->, >=stealth, thin] (1.8, 1.6) -- (1.8, 4) node[midway, fill=white] {$\ell$};
    
    % Tension labels T1 and T2
    \node[below left] at (-0.9, 2.8) {$T_1$};
    \node[above right] at (1.8, 0.8) {$T_2$};
    
    % Angle arcs for theta and phi
    % theta at C (-1.8, 4)
    \draw[thin] (-1.8, 4) +(0.4, 0) arc (0:-53.13:0.4) node[midway, below right, xshift=-1pt, yshift=2pt] {$\theta$};
    
    % phi at F (3.6, 0)
    \draw[thin] (3.6, 0) +(-0.5, 0) arc (180:156.04:0.5) node[midway, left, yshift=2pt] {$\phi$};
\end{tikzpicture}
\end{center}
\vspace{0.3em}

\begin{enumerate}
    \item[(i)] By considering horizontal and vertical components of the forces at $P$, show that
    \[
    \tan \theta = \tan \phi + \frac{Mg}{T_2 \cos \phi}.
    \]
    
    \item[(ii)] Hence, or otherwise, show that the point $P$ cannot be lifted to a position $\frac{2h}{3}$ metres above the floor.
\end{enumerate}
\end{problem}

\begin{hint}
For part (i), resolve forces horizontally and vertically at point $P$. For part (ii), use the relationship from part (i) and consider the geometry constraints involving the heights and angles.
\end{hint}
